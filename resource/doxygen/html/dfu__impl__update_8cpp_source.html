41 int dfu_impl_t::upd_txfilter (vtx_t u, const jobmeta_t &jobmeta,
42  const std::map<std::string, int64_t> &dfu)
43 {
44  // idata tag and exclusive checker update
45  int64_t span = -1;
46  planner_t *x_checker = NULL;
47 
48  // Tag on a vertex with exclusive access or all of its ancestors
49  (*m_graph)[u].idata.tags[jobmeta.jobid] = jobmeta.jobid;
50  // Update x_checker used for quick exclusivity check during matching
51  if ( (x_checker = (*m_graph)[u].idata.x_checker) == NULL) {
52  m_err_msg += __FUNCTION__;
53  m_err_msg += ": x_checker not installed.\n";
54  return -1;
55  }
56  if ( (span = planner_add_span (x_checker, jobmeta.at,
57  jobmeta.duration, 1)) == -1) {
58  m_err_msg += __FUNCTION__;
59  m_err_msg += ": planner_add_span returned -1.\n";
60  m_err_msg += strerror (errno);
61  m_err_msg += "\n";
62  return -1;
63  }
64  (*m_graph)[u].idata.x_spans[jobmeta.jobid] = span;
65  return 0;

68 int dfu_impl_t::upd_agfilter (vtx_t u, const subsystem_t &s,
69  const jobmeta_t &jobmeta,
70  const std::map<std::string, int64_t> &dfu)
71 {
72  // idata subtree aggregate prunning filter
73  planner_multi_t *subtree_plan = (*m_graph)[u].idata.subplans[s];
74  if (subtree_plan && !dfu.empty ()) {
75  int64_t span = -1;
76  std::vector<uint64_t> aggregate;
77  // Update the subtree aggregate pruning filter of this vertex
78  // using the new aggregates passed by dfu.
79  count_relevant_types (subtree_plan, dfu, aggregate);
80  span = planner_multi_add_span (subtree_plan,
81  jobmeta.at, jobmeta.duration,
82  &(aggregate[0]), aggregate.size ());
83  if (span == -1) {
84  m_err_msg += __FUNCTION__;
85  m_err_msg += ": planner_multi_add_span returned -1.\n";
86  m_err_msg += strerror (errno);
87  m_err_msg += "\n";
88  return -1;
89  }
90  (*m_graph)[u].idata.job2span[jobmeta.jobid] = span;
91  }
92  return 0;

108 int dfu_impl_t::upd_by_outedges (const subsystem_t &subsystem,
109  const jobmeta_t &jobmeta, vtx_t u, edg_t e)
110 {
111  size_t len = 0;
112  vtx_t tgt = target (e, *m_graph);
113  planner_multi_t *subplan = (*m_graph)[tgt].idata.subplans[subsystem];
114  if (subplan) {
115  if ( (len = planner_multi_resources_len (subplan)) == 0)
116  return -1;
117 
118  // Set dynamic traversing order based on the following heuristics:
119  // 1. Current-time (jobmeta.now) resource availability
120  // 2. Last pruning filter resource type (if additional
121  // pruning filter type was given, that's a good
122  // indication that it is the scarcest resource)
123  int64_t avail = planner_multi_avail_resources_at (subplan,
124  jobmeta.now, len - 1);
125  // Special case to skip (e.g., leaf resource vertices)
126  if (avail == 0 && planner_multi_span_size (subplan) == 0)
127  return 0;
128 
129  auto key = std::make_pair ((*m_graph)[e].idata.get_weight (),
130  (*m_graph)[tgt].uniq_id);
131  m_graph_db->metadata.by_outedges[u].erase (key);
132 
133  (*m_graph)[e].idata.set_weight ((avail == -1)? 0 : avail);
134  key = std::make_pair ((*m_graph)[e].idata.get_weight (),
135  (*m_graph)[tgt].uniq_id);
136  // Reinsert so that outedges are maintained according to the current
137  // resource availability state. Leverage the fact that std::map
138  // uses a RedBlack tree keep its elemented in sorted order.
139  auto ret = m_graph_db->metadata.by_outedges[u].insert (
140  std::make_pair (key, e));
141  if (!ret.second) {
142  errno = ENOMEM;
143  return -1;
144  }
145  }
146  return 0;
147 }

149 int dfu_impl_t::upd_plan (vtx_t u, const subsystem_t &s, unsigned int needs,
150  bool excl, const jobmeta_t &jobmeta, bool full,
151  int &n)
152 {
153  int rc = 0;
154 
155  if (excl) {
156 
157  n++;
158  if (!full) {
159  // If not full mode, plan has already been updated, thus return.
160  return 0;
161  }
162 
163  int64_t span = -1;
164  planner_t *plans = NULL;
165 
166  if ( (plans = (*m_graph)[u].schedule.plans) == NULL) {
167  m_err_msg += __FUNCTION__;
168  m_err_msg += ": plans not installed.\n";
169  }
170  if ( (span = planner_add_span (plans, jobmeta.at, jobmeta.duration,
171  (const uint64_t)needs)) == -1) {
172  m_err_msg += __FUNCTION__;
173  m_err_msg += ": planner_add_span returned -1.\n";
174  if (errno != 0) {
175  m_err_msg += strerror (errno);
176  m_err_msg += "\n";
177  }
178  rc = -1;
179  goto done;
180  }
181 
182  switch (jobmeta.alloc_type) {
183  case jobmeta_t::alloc_type_t::AT_ALLOC:
184  (*m_graph)[u].schedule.allocations[jobmeta.jobid] = span;
185  break;
186  case jobmeta_t::alloc_type_t::AT_ALLOC_ORELSE_RESERVE:
187  (*m_graph)[u].schedule.reservations[jobmeta.jobid] = span;
188  break;
189  case jobmeta_t::alloc_type_t::AT_SATISFIABILITY:
190  break;
191  default:
192  rc = -1;
193  errno = EINVAL;
194  break;
195  }
196  }
197 
198 done:
199  return rc;
200 }

202 int dfu_impl_t::accum_to_parent (vtx_t u, const subsystem_t &subsystem,
203  unsigned int needs, bool excl,
204  const std::map<std::string, int64_t> &dfu,
205  std::map<std::string, int64_t> &to_parent)
206 {
207  // Build up the new aggregates that will be used by subtree
208  // aggregate pruning filter. If exclusive, none of the vertex's resource
209  // is available (size). If not, all will be available (size - needs).
210  if (excl)
211  accum_if (subsystem,
212  (*m_graph)[u].type, (*m_graph)[u].size, to_parent);
213  else
214  accum_if (subsystem,
215  (*m_graph)[u].type, (*m_graph)[u].size - needs, to_parent);
216 
217  // Pass up the new subtree aggregates collected so far to the parent.
218  for (auto &kv : dfu)
219  accum_if (subsystem, kv.first, kv.second, to_parent);
220 
221  return 0;
222 }

298 int dfu_impl_t::upd_dfv (vtx_t u, std::shared_ptr<match_writers_t> &writers,
299  unsigned int needs, bool excl,
300  const jobmeta_t &jobmeta, bool full,
301  std::map<std::string, int64_t> &to_parent,
302  bool emit_shadow)
303 {
304  int n_plans = 0;
305  std::map<std::string, int64_t> dfu;
306  const std::string &dom = m_match->dom_subsystem ();
307  f_out_edg_iterator_t ei, ei_end;
308  bool mod = modify_traversal (u, emit_shadow);
309  excl = excl || mod;
310  m_trav_level++;
311  (*m_graph)[u].idata.colors[dom] = m_color.gray ();
312  for (auto &subsystem : m_match->subsystems ()) {
313  for (tie (ei, ei_end) = out_edges (u, *m_graph); ei != ei_end; ++ei) {
314  if (!in_subsystem (*ei, subsystem) || stop_explore (*ei, subsystem))
315  continue;
316  if (stop_explore_best (*ei, mod))
317  continue;
318 
319  vtx_t tgt = target (*ei, *m_graph);
320  int n_plan_sub = 0;
321  bool x = get_eff_exclusive (
322  (*m_graph)[*ei].idata.get_exclusive (), mod);
323  unsigned needs = get_eff_needs (
324  (*m_graph)[*ei].idata.get_needs (),
325  (*m_graph)[tgt].size, mod);
326 
327  if (subsystem == dom) {
328  n_plan_sub += upd_dfv (tgt, writers,
329  needs, x, jobmeta, full, dfu, mod);
330  } else {
331  n_plan_sub += upd_upv (tgt, writers, subsystem,
332  needs, x, jobmeta, full, dfu);
333  }
334 
335  if (n_plan_sub > 0) {
336  if (m_match->get_stop_on_k_matches () > 0
337  && upd_by_outedges (subsystem, jobmeta, u, *ei) < 0) {
338  m_err_msg += __FUNCTION__;
339  m_err_msg += ": upd_by_outedges returned -1.\n";
340  }
341  if (emit_edg (*ei, writers) == -1) {
342  m_err_msg += __FUNCTION__;
343  m_err_msg += ": emit_edg returned -1.\n";
344  }
345  n_plans += n_plan_sub;
346  }
347  }
348  }
349  (*m_graph)[u].idata.colors[dom] = m_color.black ();
350  return upd_sched (u, writers, dom, needs,
351  excl, n_plans, jobmeta, full, dfu, to_parent);
352 }

354 int dfu_impl_t::rem_txfilter (vtx_t u, int64_t jobid, bool &stop)
355 {
356  int rc = -1;
357  int64_t span = -1;
358  planner_t *x_checker = NULL;
359  auto &x_spans = (*m_graph)[u].idata.x_spans;
360  auto &tags = (*m_graph)[u].idata.tags;
361 
362  if (tags.find (jobid) == tags.end ()) {
363  stop = true;
364  rc = 0;
365  goto done;
366  }
367  if (x_spans.find (jobid) == x_spans.end ()) {
368  m_err_msg += __FUNCTION__;
369  m_err_msg += ": jobid isn't found in x_spans table.\n ";
370  goto done;
371  }
372 
373  x_checker = (*m_graph)[u].idata.x_checker;
374  (*m_graph)[u].idata.tags.erase (jobid);
375  span = (*m_graph)[u].idata.x_spans[jobid];
376  (*m_graph)[u].idata.x_spans.erase (jobid);
377  if ( (rc = planner_rem_span (x_checker, span)) == -1) {
378  m_err_msg += __FUNCTION__;
379  m_err_msg += "planner_rem_span returned -1.\n";
380  m_err_msg += (*m_graph)[u].name + ".\n";
381  m_err_msg += strerror (errno);
382  m_err_msg += ".\n";
383  }
384 
385 done:
386  return rc;
387 }

468 int dfu_impl_t::rem_dfv (vtx_t u, int64_t jobid)
469 {
470  int rc = 0;
471  bool stop = false;
472  const std::string &dom = m_match->dom_subsystem ();
473  f_out_edg_iterator_t ei, ei_end;
474 
475  if ( (rc = rem_idata (u, jobid, dom, stop)) != 0 || stop)
476  goto done;
477  if ( (rc = rem_plan (u, jobid)) != 0)
478  goto done;
479  for (auto &subsystem : m_match->subsystems ()) {
480  for (tie (ei, ei_end) = out_edges (u, *m_graph); ei != ei_end; ++ei) {
481  if (!in_subsystem (*ei, subsystem) || stop_explore (*ei, subsystem))
482  continue;
483  vtx_t tgt = target (*ei, *m_graph);
484  if (subsystem == dom)
485  rc += rem_dfv (tgt, jobid);
486  else
487  rc += rem_upv (tgt, jobid);
488  }
489  }
490 done:
491  return rc;

494 int dfu_impl_t::rem_exv (int64_t jobid)
495 {
496  int rc = -1;
497  int64_t span = -1;
498  vtx_iterator_t vi, v_end;
499  edg_iterator_t ei, e_end;
500  resource_graph_t &g = m_graph_db->resource_graph;
501 
502  // Exhausitive visit (exv) is required when jobid came from an allocation
503  // created by a traverser different from this traverser, for example, one
504  // that uses a different dominant subsystem. If this code is used
505  // in Flux's resource module, this condition can arise when this module
506  // is reloaded with a different match policy and the job-manager wants
507  // to reconstruct a previously allocated job.
508  // In this case, you can't find allocated resources from an accelerated
509  // depth first visit (dfv). There won't be no idata for that allocation.
510  for (boost::tie (vi, v_end) = boost::vertices (g); vi != v_end; ++vi) {
511  if (g[*vi].schedule.allocations.find (jobid)
512  != g[*vi].schedule.allocations.end ()) {
513  span = g[*vi].schedule.allocations[jobid];
514  g[*vi].schedule.allocations.erase (jobid);
515  } else if (g[*vi].schedule.reservations.find (jobid)
516  != g[*vi].schedule.reservations.end ()) {
517  span = g[*vi].schedule.reservations[jobid];
518  g[*vi].schedule.reservations.erase (jobid);
519  } else {
520  continue;
521  }
522 
523  if ( (rc += planner_rem_span (g[*vi].schedule.plans, span)) == -1) {
524  m_err_msg += __FUNCTION__;
525  m_err_msg += ": planner_rem_span returned -1.\n";
526  m_err_msg += "name=" + g[*vi].name + "uniq_id=";
527  m_err_msg += std::to_string (g[*vi].uniq_id) + ".\n";
528  m_err_msg += strerror (errno);
529  m_err_msg += ".\n";
530  }
531  }
532 
533  return (!rc)? 0 : -1;
534 }

543 int dfu_impl_t::update (vtx_t root, std::shared_ptr<match_writers_t> &writers,
544  jobmeta_t &jobmeta)
545 {
546  int rc = -1;
547  std::map<std::string, int64_t> dfu;
548  const std::string &dom = m_match->dom_subsystem ();
549 
550  if (m_graph_db->metadata.v_rt_edges[dom].get_trav_token ()
551  != m_best_k_cnt) {
552  m_err_msg += __FUNCTION__;
553  m_err_msg += ": resource state wasn't properly set up for update.\n";
554  return -1;
555  }
556 
557  unsigned int excl = m_graph_db->metadata.v_rt_edges[dom].get_exclusive ();
558  bool x = (excl == 0)? false : true;
559  unsigned int needs = m_graph_db->metadata.v_rt_edges[dom].get_needs ();
560  m_color.reset ();
561 
562  bool emit_shadow = modify_traversal (root, false);
563  if ((rc = upd_dfv (root, writers, needs,
564  x, jobmeta, true, dfu, emit_shadow)) > 0) {
565  uint64_t starttime = jobmeta.at;
566  uint64_t endtime = jobmeta.at + jobmeta.duration;
567  if (writers->emit_tm (starttime, endtime) == -1) {
568  m_err_msg += __FUNCTION__;
569  m_err_msg += ": emit_tm returned -1.\n";
570  }
571  if (jobmeta.is_queue_set ()) {
572  if (writers->emit_attrs ("queue", jobmeta.get_queue ()) == -1) {
573  m_err_msg += __FUNCTION__;
574  m_err_msg += ": emit_attrs returned -1.\n";
575  }
576  }
577  }
578 
579  return (rc > 0)? 0 : -1;
580 }
