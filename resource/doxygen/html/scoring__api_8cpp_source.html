36 void scoring_api_t::handle_new_subsystem (const subsystem_t &s)
37 {
38  if (m_ssys_map.find (s) == m_ssys_map.end ()) {
39  auto o = new std::map<const std::string, detail::evals_t *>();
40  m_ssys_map.insert (std::make_pair (s, o));

44 void scoring_api_t::handle_new_resrc_type (const subsystem_t &s,
45  const std::string &r)
46 {
47  if (m_ssys_map[s]->find (r) == m_ssys_map[s]->end ()) {
48  auto e = new detail::evals_t (r);

50  }

65 scoring_api_t::scoring_api_t (const scoring_api_t &o)
66 {
67  for (auto &p : o.m_ssys_map) {
68  const subsystem_t &s = p.first;
69  auto obj = new std::map<const std::string, detail::evals_t *>();
70  m_ssys_map.insert (std::make_pair (s, obj));
71  auto &tmap = *(p.second);
72  for (auto &p2 : tmap) {
73  const std::string &res_type = p2.first;
74  detail::evals_t *ne = new detail::evals_t ();
75  *ne = *(p2.second);
76  (*m_ssys_map[s]).insert (std::make_pair (res_type, ne));
77  }
78  }
79 }

98 scoring_api_t::~scoring_api_t ()
99 {
100  auto i = m_ssys_map.begin ();
101  while (i != m_ssys_map.end ()) {
102  auto tmap = i->second;
103  auto j = tmap->begin ();
104  while (j != tmap->end ()) {
105  delete j->second;
106  j = tmap->erase (j);
107  }
108  delete i->second;
109  i = m_ssys_map.erase (i);
110  }
111 }

215 void scoring_api_t::merge (const scoring_api_t &o)
216 {
217  for (auto &kv : o.m_ssys_map) {
218  const subsystem_t &s = kv.first;
219  auto &tmap = *(kv.second);
220  for (auto &kv2 : tmap) {
221  const std::string &r = kv2.first;
222  auto &ev = *(kv2.second);
223  handle_new_keys (s, r);
224  auto res_evals = (*m_ssys_map[s])[r];
225  res_evals->merge (ev);
226  }
227  }
