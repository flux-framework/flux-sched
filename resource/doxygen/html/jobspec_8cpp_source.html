46 void parse_yaml_count (Resource& res, const YAML::Node &cnode)
47 {
48  /* count can have an unsigned integer value */
49  if (cnode.IsScalar()) {
50  res.count.min = cnode.as<unsigned>();
51  res.count.max = res.count.min;
52  return;
53  }
54 
55  /* or count may be a more complicated verbose form */
56  if (!cnode.IsMap()) {
57  throw parse_error (cnode, "count is not a mapping");
58  }
59 
60  /* Verify existence of required entries */
61  if (!cnode["min"]) {
62  throw parse_error (cnode, "Key \"min\" missing from count");
63  }
64  if (!cnode["min"].IsScalar()) {
65  throw parse_error (cnode["min"], "Value of \"min\" must be a scalar");
66  }
67  if (cnode["max"] && !cnode["max"].IsScalar()) {
68  throw parse_error (cnode["max"], "Value of \"max\" must be a scalar");
69  }
70  if (cnode["operator"] && !cnode["operator"].IsScalar()) {
71  throw parse_error (cnode["operator"],
72  "Value of \"operator\" must be a scalar");
73  }
74  if (cnode["operand"] && !cnode["operand"].IsScalar()) {
75  throw parse_error (cnode["operand"],
76  "Value of \"operand\" must be a scalar");
77  }
78 
79  /* Validate values of entries */
80  res.count.min = cnode["min"].as<unsigned>();
81  if (cnode["max"]) {
82  res.count.max = cnode["max"].as<unsigned>();
83  }
84  if (cnode["operator"]) {
85  res.count.oper = cnode["operator"].as<char>();
86  }
87  if (cnode["operand"]) {
88  res.count.operand = cnode["operand"].as<int>();
89  }
90 
91  if (res.count.min < 1) {
92  throw parse_error (cnode["min"], "\"min\" must be greater than zero");
93  }
94  if (res.count.max < 1) {
95  throw parse_error (cnode["max"], "\"max\" must be greater than zero");
96  }
97  if (res.count.max < res.count.min) {
98  throw parse_error (cnode["max"],
99  "\"max\" must be greater than or equal to \"min\"");
100  }
101  switch (res.count.oper) {
102  case '+':
103  case '*':
104  case '^':
105  break;
106  default:
107  throw parse_error (cnode["operator"], "Invalid count operator");
108  }
109 }
110 }

116 Resource::Resource (const YAML::Node &resnode)
117 {
118  unsigned field_count = 0;
119 
120  /* The resource must be a mapping */
121  if (!resnode.IsMap()) {
122  throw parse_error (resnode, "resource is not a mapping");
123  }
124  if (!resnode["type"]) {
125  throw parse_error (resnode, "Key \"type\" missing from resource");
126  }
127  if (!resnode["type"].IsScalar()) {
128  throw parse_error (resnode["type"],
129  "Value of \"type\" must be a scalar");
130  }
131  type = resnode["type"].as<std::string>();
132  field_count++;
133 
134  if (!resnode["count"]) {
135  throw parse_error (resnode, "Key \"count\" missing from resource");
136  }
137  parse_yaml_count (*this, resnode["count"]);
138  field_count++;
139 
140  if (resnode["unit"]) {
141  if (!resnode["unit"].IsScalar()) {
142  throw parse_error (resnode["unit"],
143  "Value of \"unit\" must be a scalar");
144  }
145  field_count++;
146  unit = resnode["unit"].as<std::string>();
147  }
148  if (resnode["exclusive"]) {
149  if (!resnode["exclusive"].IsScalar()) {
150  throw parse_error (resnode["exclusive"],
151  "Value of \"exclusive\" must be a scalar");
152  }
153  field_count++;
154  std::string val = resnode["exclusive"].as<std::string>();
155  if (val == "false") {
156  exclusive = tristate_t::FALSE;
157  } else if (val == "true") {
158  exclusive = tristate_t::TRUE;
159  } else {
160  throw parse_error (resnode["exclusive"],
161  "Value of \"exclusive\" must be either \"true\" or \"false\"");
162  }
163  }
164 
165  if (resnode["with"]) {
166  field_count++;
167  with = parse_yaml_resources (resnode["with"]);
168  }
169 
170  if (resnode["label"]) {
171  if (!resnode["label"].IsScalar()) {
172  throw parse_error (resnode["label"],
173  "Value of \"label\" must be a scalar");
174  }
175  field_count++;
176  label = resnode["label"].as<std::string>();
177  } else if (type == "slot") {
178  throw parse_error (resnode, "All slots must be labeled");
179  }
180 
181  if (resnode["id"]) {
182  if (!resnode["id"].IsScalar()) {
183  throw parse_error (resnode["id"],
184  "Value of \"id\" must be a scalar");
185  }
186  field_count++;
187  id = resnode["id"].as<std::string>();
188  }
189 
190  if (field_count != resnode.size()) {
191  throw parse_error (resnode, "Unrecognized key in resource mapping");
192  }
193 
194  if (resnode.size() < 2 || resnode.size() > 10) {
195  throw parse_error (resnode,
196  "impossible number of entries in resource mapping");
197  }

201 Task::Task (const YAML::Node &tasknode)
202 {
203  /* The task node must be a mapping */
204  if (!tasknode.IsMap()) {
205  throw parse_error (tasknode, "task is not a mapping");
206  }
207  if (!tasknode["command"]) {
208  throw parse_error (tasknode, "Key \"command\" missing from task");
209  }
210  if (tasknode["command"].IsSequence()) {
211  command = tasknode["command"].as<std::vector<std::string>>();
212  } else {
213  throw parse_error (tasknode["command"],
214  "\"command\" value must be a sequence");
215  }
216 
217  /* Import slot */
218  if (!tasknode["slot"]) {
219  throw parse_error (tasknode, "Key \"slot\" missing from task");
220  }
221  if (!tasknode["slot"].IsScalar()) {
222  throw parse_error (tasknode["slot"],
223  "Value of task \"slot\" must be a YAML scalar");
224  }
225  slot = tasknode["slot"].as<std::string>();
226 
227  /* Import count mapping */
228  if (tasknode["count"]) {
229  YAML::Node count_node = tasknode["count"];
230  if (!count_node.IsMap()) {
231  throw parse_error (count_node,
232  "\"count\" in task is not a mapping");
233  }
234  for (auto&& entry : count_node) {
235  count[entry.first.as<std::string>()]
236  = entry.second.as<std::string>();
237  }
238  }
239 
240  /* Import distribution if it is present */
241  if (tasknode["distribution"]) {
242  if (!tasknode["distribution"].IsScalar()) {
243  throw parse_error (tasknode["distribution"],
244  "Value of task \"distribution\" must be a YAML scalar");
245  }
246  distribution = tasknode["distribution"].as<std::string>();
247  }
248 
249  /* Import attributes mapping if it is present */
250  if (tasknode["attributes"]) {
251  YAML::Node attrs = tasknode["attributes"];
252  if (!attrs.IsMap()) {
253  throw parse_error (attrs, "\"attributes\" in task is not a mapping");
254  }
255  for (auto&& attr : attrs) {
256  attributes[attr.first.as<std::string>()]
257  = attr.second.as<std::string>();
258  }
259  }
260 
261  if (tasknode.size() < 3 || tasknode.size() > 5) {
262  throw parse_error (tasknode,
263  "impossible number of entries in task mapping");
264  }
265 }

304 Attributes parse_yaml_attributes (const YAML::Node &attrs)
305 {
306  Attributes a;
307 
308  if (!attrs.IsMap ()) {
309  throw parse_error (attrs, "\"attributes\" is not a map");
310  }
311  for (auto&& kv : attrs) {
312  if (kv.first.as<std::string>() == "user") {
313  a.user = kv.second;
314  }
315  else if (kv.first.as<std::string>() == "system") {
316  for (auto&& s : kv.second) {
317  if (s.first.as<std::string>() == "duration") {
318  a.system.duration = s.second.as<double>();
319  }
320  else if (s.first.as<std::string>() == "queue") {
321  a.system.queue = s.second.as<std::string>();
322  }
323  else if (s.first.as<std::string>() == "cwd") {
324  a.system.cwd = s.second.as<std::string>();
325  }
326  else if (s.first.as<std::string>() == "environment") {
327  for (auto&& e : s.second) {
328  a.system.environment[e.first.as<std::string>()]
329  = e.second.as<std::string>();
330  }
331  }
332  else if (s.first.as<std::string>() == "constraints") {
333  a.system.constraint = constraint_parser (s.second);
334  } else {
335  a.system.optional[s.first.as<std::string>()] = s.second;
336  }
337  }
338  }
339  else {
340  throw parse_error (kv.second, "Unknown key in \"attributes\"");
341  }
342  }
343  return a;
344 }

347 Jobspec::Jobspec(const YAML::Node &top)
348 {
349  try {
350  /* The top yaml node of the jobspec must be a mapping */
351  if (!top.IsMap()) {
352  throw parse_error (top, "Top level of jobspec is not a mapping");
353  }
354  /* The four keys must be the following */
355  if (!top["version"]) {
356  throw parse_error (top, "Missing key \"version\" in top level mapping");
357  }
358  if (!top["resources"]) {
359  throw parse_error (top, "Missing key \"resource\" in top level mapping");
360  }
361  if (!top["tasks"]) {
362  throw parse_error (top, "Missing key \"tasks\" in top level mapping");
363  }
364  if (!top["attributes"]) {
365  throw parse_error (top, "Missing key \"attributes\" in top level mapping");
366  }
367  /* There must be exactly four entries in the mapping */
368  if (top.size() != 4) {
369  throw parse_error (top, "Top mapping in jobspec must have exactly four entries");
370  }
371 
372  /* Import version */
373  if (!top["version"].IsScalar()) {
374  throw parse_error (top["version"],
375  "\"version\" must be an unsigned integer");
376  }
377  version = top["version"].as<unsigned int>();
378  if (version < 1 || version > 9999) {
379  throw parse_error (top["version"],
380  "Only jobspec \"version\" 1-9999 is supported");
381  }
382 
383  /* Import attributes mappings */
384  if (!top["attributes"].IsNull ())
385  attributes = parse_yaml_attributes (top["attributes"]);
386 
387  /* Import resources section */
388  resources = parse_yaml_resources (top["resources"]);
389 
390  /* Import tasks section */
391  tasks = parse_yaml_tasks (top["tasks"]);
392  } catch (YAML::Exception& e) {
393  throw parse_error(e.what());
394  }
395 }

416 /*
417  * This class magically makes everything in 'dest' stream
418  * indented as long as this class is in scope. Once it
419  * it goes out of scope and is destroyed, the indenting
420  * disappears.
421  */
422 class IndentingOStreambuf : public std::streambuf
423 {
424  std::streambuf* myDest;
425  bool myIsAtStartOfLine;
426  std::string myIndent;
427  std::ostream* myOwner;
428 protected:
429  virtual int overflow( int ch )
430  {
431  if (myIsAtStartOfLine && ch != '\n') {
432  myDest->sputn (myIndent.data(), myIndent.size());
433  }
434  myIsAtStartOfLine = ch == '\n';
435  return myDest->sputc (ch);
436  }
437 public:
438  explicit IndentingOStreambuf (std::streambuf* dest, int indent = 2)
439  : myDest (dest)
440  , myIsAtStartOfLine (true)
441  , myIndent (indent, ' ')
442  , myOwner(NULL)
443  {
444  }
445  explicit IndentingOStreambuf (std::ostream& dest, int indent = 2)
446  : myDest (dest.rdbuf ())
447  , myIsAtStartOfLine (true)
448  , myIndent (indent, ' ')
449  , myOwner (&dest)
450  {
451  myOwner->rdbuf (this);
452  }
453  virtual ~IndentingOStreambuf()
454  {
455  if ( myOwner != NULL ) {
456  myOwner->rdbuf (myDest);
457  }
458  }
459 };
460 }
