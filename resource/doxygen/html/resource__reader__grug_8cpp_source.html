75 int dfs_emitter_t::path_prefix (const std::string &path,
76  int uplevel, std::string &prefix)
77 {
78  size_t pos = 0;
79  unsigned int occurrence = 0;
80  auto num_slashes = count (path.begin (), path.end (), '/');
81  if (uplevel >= num_slashes)
82  return -1;
83  while (occurrence != (num_slashes - uplevel + 1)) {
84  pos = path.find ("/", pos);
85  if (pos == std::string::npos)
86  break;
87  pos += 1;
88  occurrence++;
89  }
90  std::string new_prefix = path.substr (0, pos);
91  if (new_prefix.back () != '/')
92  new_prefix.push_back ('/');
93  prefix = std::move (new_prefix);
94  return 0;
95 }

97 //
98 // This id is local to its ancestor level defined by "scope"
99 // scope=0: id is local to its parent
100 // scope=1: id is local to its grand parent
101 // For example, in rack[1]->node[18]->socket[2]->core[8] configuration,
102 // if scope is 1, the id space of a core resource is local to
103 // the node level instead of the socket level.
104 // So, 16 cores in each node will have 0-15, instead of repeating
105 // 0-7 and 0-7, which will be the case if the scope is 0.
106 //
107 int dfs_emitter_t::gen_id (gge_t e, const gg_t &recipe, int i, int sz, int j)
108 {
109  int h = 0;
110  int j_dim_wrap = 1;
111  int scope = recipe[e].id_scope;
112 
113  if (scope < 0)
114  return -1;
115  else if (scope == 0)
116  return recipe[e].id_start + i;
117 
118  if (scope > (int) m_hier_scales.size ())
119  scope = m_hier_scales.size ();
120  j_dim_wrap = 1;
121  std::deque<int>::const_iterator iter;
122  for (h = 0; h < scope; ++h)
123  j_dim_wrap *= m_hier_scales[h];
124 
125  return recipe[e].id_start
126  + (j % j_dim_wrap * sz * recipe[e].id_stride)
127  + (i * recipe[e].id_stride);
128 }

130 int dfs_emitter_t::raw_edge (vtx_t src_v, vtx_t tgt_v, edg_t &e)
131 {
132  bool inserted;
133  out_edg_iterator_t ei, ee;
134  resource_graph_t &g = *m_g_p;
135 
136  tie (ei, ee) = out_edges (src_v, g);
137  for ( ; ei != ee; ++ei) {
138  if (target (*ei, g) == tgt_v) {
139  e = (*ei);
140  return 0;
141  }
142  }
143  tie (e, inserted) = add_edge (src_v, tgt_v, g);
144  if (!inserted) {
145  errno = ENOMEM;
146  m_err_msg += "error inserting a new edge: "
147  + g[src_v].name + " -> " + g[tgt_v].name + "; ";
148  return -1;
149  }
150  // add this edge to by_outedges metadata
151  auto iter = m_gm_p->by_outedges.find (src_v);
152  if (iter == m_gm_p->by_outedges.end ()) {
153  auto ret = m_gm_p->by_outedges.insert (
154  std::make_pair (
155  src_v,
156  std::map<std::pair<uint64_t, int64_t>, edg_t,
157  std::greater<std::pair<uint64_t,
158  int64_t>>> ()));
159  if (!ret.second) {
160  errno = ENOMEM;
161  m_err_msg += "error creating out-edge metadata map: "
162  + g[src_v].name + " -> " + g[tgt_v].name + "; ";
163  return -1;
164  }
165  iter = m_gm_p->by_outedges.find (src_v);
166  }
167  // Use a temporary for readability
168  std::pair<uint64_t, int64_t> key = std::make_pair (
169  g[e].idata.get_weight (),
170  g[tgt_v].uniq_id);
171  auto ret = iter->second.insert (std::make_pair (key, e));
172  if (!ret.second) {
173  errno = ENOMEM;
174  m_err_msg += "error inserting an edge into out-edge metadata map:"
175  + g[src_v].name + " -> " + g[tgt_v].name + "; ";
176  return -1;
177  }
178  return 0;
179 }

181 void dfs_emitter_t::emit_edges (gge_t ge, const gg_t &recipe,
182  vtx_t src_v, vtx_t tgt_v)
183 {
184  edg_t e;
185  int rc = 0;
186  resource_graph_t &g = *m_g_p;
187  if ( (rc = raw_edge (src_v, tgt_v, e)) < 0)
188  return;
189  g[e].idata.member_of[recipe[ge].e_subsystem]
190  = recipe[ge].relation;
191  g[e].name[recipe[ge].e_subsystem] = recipe[ge].relation;
192  if ( (rc = raw_edge (tgt_v, src_v, e)) < 0)
193  return;
194  g[e].idata.member_of[recipe[ge].e_subsystem]
195  = recipe[ge].rrelation;
196  g[e].name[recipe[ge].e_subsystem] = recipe[ge].rrelation;
197 }

199 vtx_t dfs_emitter_t::emit_vertex (ggv_t u, gge_t e, const gg_t &recipe,
200  vtx_t src_v, int i, int sz, int j)
201 {
202  resource_graph_t &g = *m_g_p;
203  resource_graph_metadata_t &m = *m_gm_p;
204  if (src_v == boost::graph_traits<resource_graph_t>::null_vertex ())
205  if (m.roots.find (recipe[u].subsystem) != m.roots.end ())
206  return (m.roots)[recipe[u].subsystem];
207 
208  vtx_t v = add_vertex (g);;
209  std::string pref = "";
210  std::string ssys = recipe[u].subsystem;
211  int id = 0;
212 
213  if (src_v == boost::graph_traits<resource_graph_t>::null_vertex ()) {
214  // ROOT vertex of graph
215  m.roots.emplace (recipe[u].subsystem, v);
216  m.v_rt_edges.emplace (recipe[u].subsystem, relation_infra_t ());
217  id = 0;
218  } else {
219  id = gen_id (e, recipe, i, sz, j);
220  pref = g[src_v].paths[ssys];
221  }
222 
223  std::string istr = (id != -1)? std::to_string (id) : "";
224  g[v].type = recipe[u].type;
225  g[v].basename = recipe[u].basename;
226  g[v].size = recipe[u].size;
227  g[v].unit = recipe[u].unit;
228  g[v].schedule.plans = planner_new (0, INT64_MAX,
229  recipe[u].size, recipe[u].type.c_str ());
230  g[v].idata.x_checker = planner_new (0, INT64_MAX,
231  X_CHECKER_NJOBS, X_CHECKER_JOBS_STR);
232  g[v].id = id;
233  g[v].name = recipe[u].basename + istr;
234  g[v].paths[ssys] = pref + "/" + g[v].name;
235  g[v].idata.member_of[ssys] = "*";
236  g[v].uniq_id = v;
237  g[v].rank = m_rank;
238 
239  //
240  // Indexing for fast look-up...
241  //
242  m.by_path[g[v].paths[ssys]].push_back (v);
243  m.by_type[g[v].type].push_back (v);
244  m.by_name[g[v].name].push_back (v);
245  m.by_rank[m_rank].push_back (v);
246  return v;
247 }

293 /*
294  * Visitor method that is invoked on a tree-edge event
295  * generated by depth_first_walk ()
296  *
297  * \param e resource generator graph edge descriptor
298  * \param recipe resource generator recipe graph
299  */
300 void dfs_emitter_t::tree_edge (gge_t e, const gg_t &recipe)
301 {
302  vtx_t src_vtx, tgt_vtx;
303  ggv_t src_ggv = source (e, recipe);
304  ggv_t tgt_ggv = target (e, recipe);
305  std::vector<vtx_t>::iterator src_it, tgt_it;
306  resource_graph_t &g = *m_g_p;
307  resource_graph_metadata_t &m = *m_gm_p;
308  std::string in;
309  int i = 0, j = 0;;
310 
311  if (recipe[src_ggv].root) {
313  if (m_gen_src_vtx[src_ggv].empty ()) {
314  vtx_t null_v = boost::graph_traits<resource_graph_t>::null_vertex ();
315  m_gen_src_vtx[src_ggv].push_back (emit_vertex (src_ggv, e, recipe,
316  null_v, 0, 1, 0));
317  }
318  }
319 
320  m_gen_src_vtx[tgt_ggv] = std::vector<vtx_t>();
321 
322  switch (m_gspec_p->to_gen_method_t (recipe[e].gen_method)) {
323  case MULTIPLY:
324  for (src_it = m_gen_src_vtx[src_ggv].begin ();
325  src_it != m_gen_src_vtx[src_ggv].end (); src_it++, j++) {
326 
327  src_vtx = *src_it;
328  for (i = 0; i < recipe[e].multi_scale; ++i) {
329  tgt_vtx = emit_vertex (tgt_ggv, e, recipe, src_vtx, i,
330  recipe[e].multi_scale, j);
331  emit_edges (e, recipe, src_vtx, tgt_vtx);
332  // TODO: Next gen src vertex; where do you clear them?
333  m_gen_src_vtx[tgt_ggv].push_back (tgt_vtx);
334  }
335  }
336  m_hier_scales.push_front (recipe[e].multi_scale);
337  break;
338 
339  case ASSOCIATE_IN:
340  for (src_it = m_gen_src_vtx[src_ggv].begin ();
341  src_it != m_gen_src_vtx[src_ggv].end (); src_it++) {
342 
343  src_vtx = *src_it;
344  for (tgt_it = m.by_type[recipe[tgt_ggv].type].begin ();
345  tgt_it != m.by_type[recipe[tgt_ggv].type].end (); tgt_it++) {
346  tgt_vtx = (*tgt_it);
347  g[tgt_vtx].paths[recipe[e].e_subsystem]
348  = g[src_vtx].paths[recipe[e].e_subsystem]
349  + "/" + g[tgt_vtx].name;
350  const std::string &tp = g[tgt_vtx].paths[recipe[e].e_subsystem];
351  m.by_path[tp].push_back (tgt_vtx);
352  g[tgt_vtx].idata.member_of[recipe[e].e_subsystem]
353  = "*";
354  emit_edges (e, recipe, src_vtx, tgt_vtx);
355  m_gen_src_vtx[tgt_ggv].push_back (tgt_vtx);
356  }
357  }
358  break;
359 
360  case ASSOCIATE_BY_PATH_IN:
361  in = recipe[e].as_tgt_subsystem;
362  for (src_it = m_gen_src_vtx[src_ggv].begin ();
363  src_it != m_gen_src_vtx[src_ggv].end (); src_it++) {
364 
365  src_vtx = *src_it;
366  for (tgt_it = m.by_type[recipe[tgt_ggv].type].begin ();
367  tgt_it != m.by_type[recipe[tgt_ggv].type].end (); tgt_it++) {
368  std::string comp_pth1, comp_pth2;
369  tgt_vtx = (*tgt_it);
370  path_prefix (g[tgt_vtx].paths[in],
371  recipe[e].as_tgt_uplvl, comp_pth1);
372  path_prefix (g[src_vtx].paths[in],
373  recipe[e].as_src_uplvl, comp_pth2);
374 
375  if (comp_pth1 != comp_pth2)
376  continue;
377 
378  g[tgt_vtx].paths[recipe[e].e_subsystem]
379  = g[src_vtx].paths[recipe[e].e_subsystem]
380  + "/" + g[tgt_vtx].name;
381  const std::string &tp = g[tgt_vtx].paths[recipe[e].e_subsystem];
382  m.by_path[tp].push_back (tgt_vtx);
383  g[tgt_vtx].idata.member_of[recipe[e].e_subsystem]
384  = "*";
385  emit_edges (e, recipe, src_vtx, tgt_vtx);
386  m_gen_src_vtx[tgt_ggv].push_back (tgt_vtx);
387  }
388  }
389  break;
390 
391  case GEN_UNKNOWN:
392  default:
393  m_err_msg += "unknown generation method; ";
394  break;
395  }
396 }
