44 struct match_perf_t {
45  double load = 0.0; /* Graph load time */
46  /* Graph uptime in seconds */
47  std::chrono::time_point<std::chrono::system_clock> graph_uptime
48  = std::chrono::system_clock::now ();
49  /* Time since stats were last cleared */
50  std::chrono::time_point<std::chrono::system_clock> time_since_reset
51  = std::chrono::system_clock::now ();
52  struct time_stats {
53  void update_stats (double elapsed) {
54  /* Update using Welford's algorithm */
55  double delta = 0.0;
56  double delta2 = 0.0;
57  njobs++;
58  njobs_reset++;
59  min = (min > elapsed)? elapsed : min;
60  max = (max < elapsed)? elapsed : max;
61  // Welford's online algorithm for variance
62  delta = elapsed - avg;
63  avg += delta / (double)njobs;
64  delta2 = elapsed - avg;
65  M2 += delta * delta2;
66  }
67  uint64_t njobs = 0; /* Total match count */
68  uint64_t njobs_reset = 0; /* Jobs since match count reset */
69  double min = std::numeric_limits<double>::max (); /* Min match time */
70  double max = 0.0; /* Max match time */
71  double avg = 0.0; /* Average match time */
72  double M2 = 0.0; /* Welford's algorithm */
73  };
74  time_stats succeeded;
75  time_stats failed;
76 };

652  * contained in it as well as r_lite key.
653  */
654 static int unpack_resources (json_t *resobj, struct idset **idset,
655  json_t **r_lite_p, json_t **jgf_p,
656  graph_duration_t &duration)
657 {
658  int rc = 0;
659  struct idset *ids;
660  int version;
661  double start = 0.0, end = 0.0;
662  json_t *r_lite = NULL;
663  json_t *jgf = NULL;
664  size_t index;
665  json_t *val;
666 
667  if (!(ids = idset_create (0, IDSET_FLAG_AUTOGROW)))
668  return -1;
669  if (resobj) {
670  if (json_unpack (resobj,
671  "{s:i s:{s:o s?F s?F} s?:o}",
672  "version", &version,
673  "execution",
674  "R_lite", &r_lite,
675  "starttime", &start,
676  "expiration", &end,
677  "scheduling", &jgf) < 0)
678  goto inval;
679  // flux-core validates these numbers, but checking here
680  // in case Fluxin is plugged into another resource manager
681  if (version != 1)
682  goto inval;
683  if (start != 0 && (start == end))
684  goto inval;
685  if (start < 0.0 || end < 0.0)
686  goto inval;
687  if (start > end)
688  goto inval;
689  // Greatest lower bound is int64_t in the rest of Fluxion
690  if (start > static_cast<double> (std::numeric_limits<int64_t>::max ())
691  || end > static_cast<double>
692  (std::numeric_limits<int64_t>::max ()))
693  goto inval;
694  // Ensure start and end are representable in system clock
695  if (start > static_cast<double>
696  (std::chrono::duration_cast<std::chrono::seconds>
697  (std::chrono::time_point<std::chrono::system_clock>::max ().time_since_epoch ()).count ())
698  || end > static_cast<double>
699  (std::chrono::duration_cast<std::chrono::seconds>
700  (std::chrono::time_point<std::chrono::system_clock>::max ().time_since_epoch ()).count ()))
701  goto inval;
702  // Expects int type argument
703  duration.graph_start = std::chrono::system_clock::from_time_t (static_cast<int64_t> (start));
704  duration.graph_end = std::chrono::system_clock::from_time_t (static_cast<int64_t> (end));
705  // Ensure there is no overflow in system clock representation
706  // (should be handled by previous check).
707  if (std::chrono::duration_cast<std::chrono::seconds>
708  (duration.graph_start.time_since_epoch ()).count () < 0
709  || std::chrono::duration_cast<std::chrono::seconds>
710  (duration.graph_end.time_since_epoch ()).count () < 0) {
711  goto inval;
712  }
713  json_array_foreach (r_lite, index, val) {
714  struct idset *r_ids;
715  unsigned long id;
716  const char *rank;
717 
718  if (json_unpack (val, "{s:s}", "rank", &rank) < 0)
719  goto inval;
720  if (!(r_ids = idset_decode (rank)))
721  goto error;
722  id = idset_first (r_ids);
723  while (id != IDSET_INVALID_ID) {
724  if (idset_set (ids, id) < 0) {
725  idset_destroy (r_ids);
726  goto error;
727  }
728  id = idset_next (r_ids, id);
729  }
730  idset_destroy (r_ids);
731  }
732  }
733  *idset = ids;
734  *r_lite_p = r_lite;
735  *jgf_p = jgf;
736  return 0;
737 inval:
738  errno = EINVAL;
739 error:
740  idset_destroy (ids);
741  return -1;
742 }

744 static int unpack_resobj (json_t *resobj,
745  std::map<const distinct_range_t,
746  std::shared_ptr<resobj_t>> &out)
747 {
748  if (!resobj)
749  goto inval;
750  try {
751  size_t index;
752  json_t *val;
753 
754  json_array_foreach (resobj, index, val) {
755  std::string range;
756  std::istringstream istr;
757  const char *rank = NULL, *core = NULL, *gpu = NULL;
758  if (json_unpack (val, "{s:s s:{s?:s s?:s}}",
759  "rank", &rank,
760  "children",
761  "core", &core,
762  "gpu", &gpu) < 0)
763  goto inval;
764  // Split the rank idset in resobj, convert each entry into
765  // distict_range_t and use it as the key to std::map.
766  // The value is the pointer to resobj_t.
767  // The distinct_range_t class provides ordering logic such that
768  // you will be able to iterate through ranges in strictly
769  // ascending order.
770  istr.str (rank);
771  while (std::getline (istr, range, ',')) {
772  uint64_t low, high;
773  std::pair<std::map<const distinct_range_t,
774  std::shared_ptr<resobj_t>>::iterator, bool> res;
775  std::shared_ptr<resobj_t> robj = std::make_shared<resobj_t> ();
776  if (distinct_range_t::get_low_high (range, low, high) < 0)
777  goto error;
778  robj->exec_target_range = range;
779  if (core && expand_ids (core, robj->core) < 0)
780  goto error;
781  if (gpu && expand_ids (gpu, robj->gpu) < 0)
782  goto error;
783  res = out.insert (std::pair<const distinct_range_t,
784  std::shared_ptr<resobj_t>> (
785  distinct_range_t{low, high}, robj));
786  if (res.second == false) {
787  errno = EEXIST;
788  goto error;
789  }
790  }
791  }
792  return 0;
793  } catch (std::bad_alloc &) {
794  errno = ENOMEM;
795  goto error;
796  } catch (std::invalid_argument &) {
797  goto inval;
798  }
799 inval:
800  errno = EINVAL;
801 error:
802  return -1;
803 }

826 static int remap_jgf_namespace (std::shared_ptr<resource_ctx_t> &ctx,
827  json_t *resobj, json_t *p_resobj)
828 {
829  size_t i, j;
830  uint64_t cur_rank = 0;
831  std::map<const distinct_range_t, std::shared_ptr<resobj_t>> robjs, p_robjs;
832 
833  if (unpack_resobj (resobj, robjs) < 0)
834  goto error;
835  if (unpack_resobj (p_resobj, p_robjs) < 0)
836  goto error;
837 
838  try {
839  // Add remap rule for rank and compute core Ids. JGF reader only
840  // needs to remap them (e.g., GPU uses Id in global scope).
841  // This loop iterates through the distinct rank ranges of R_lite in
842  // the parent's namespace in ascending order.
843  // We first translate the ascending rank range sequence into
844  // 0-based monotonically increasing rank range sequence, for example
845  // A: <"3", "5", "6", "20-21"> --> B: <"1", "2", "3", "4-5">.
846  // Now, per core resource.acquire's remapping rule, for each range found
847  // in the parent namespace's R_lite, you should be able to find a
848  // corresponding range in the current namespace's R_lite.
849  // The current R_lite's range sequence for the above example might
850  // be C: <"1-2", "3", "4-5">. Then, B's "1" and "2" will be resolved
851  // to C's "1-2" range. B's "3" to C's "3". And B's "4-5" to C's "4-5".
852  for (auto kv : p_robjs) {
853  distinct_range_t new_range{cur_rank,
854  kv.first.get_high ()
855  - kv.first.get_low () + cur_rank};
856  std::shared_ptr<resobj_t> entry = nullptr;
857  std::shared_ptr<resobj_t> p_entry = kv.second;
858  if (robjs.find (new_range) == robjs.end ()) {
859  // Can't find the rank range in the current namespace
860  // corresponding to the (remapped) range of the parent namespace!
861  errno = ENOENT;
862  goto error;
863  }
864  entry = robjs[new_range];
865  if (ctx->reader->namespace_remapper.add_exec_target_range (
866  p_entry->exec_target_range,
867  new_range) < 0)
868  goto error;
869  for (j = 0; j < p_entry->core.size (); j++) {
870  if (ctx->reader->namespace_remapper.add (
871  p_entry->exec_target_range, "core",
872  p_entry->core[j], entry->core[j]) < 0)
873  goto error;
874  }
875  cur_rank += (kv.first.get_high () - kv.first.get_low () + 1);
876  }
877  } catch (std::invalid_argument &) {
878  errno = EINVAL;
879  goto error;
880  }
881  return 0;
882 error:
883  return -1;
884 }
