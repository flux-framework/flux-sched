33 planner_multi::planner_multi (int64_t base_time, uint64_t duration,
34  const uint64_t *resource_totals,
35  const char **resource_types, size_t len)
36 {
37  size_t i = 0;
38  std::string type;
39  planner_t *p = nullptr;
40 
41  m_iter.on_or_after = 0;
42  m_iter.duration = 0;
43  for (i = 0; i < len; ++i) {
44  try {
45  type = std::string (resource_types[i]);
46  p = new planner_t (base_time, duration,
47  resource_totals[i],
48  resource_types[i]);
49  } catch (std::bad_alloc &e) {
50  errno = ENOMEM;
51  throw std::bad_alloc ();
52  }
53  m_iter.counts[type] = 0;
54  m_types_totals_planners.push_back ({type, resource_totals[i], p});
55  }
56  m_span_counter = 0;
57 }

59 planner_multi::planner_multi (const planner_multi &o)
60 {
61  for (auto iter : o.m_types_totals_planners) {
62  planner_t *np = nullptr;
63  if (iter.planner) {
64  try {
65  np = new planner_t (*(iter.planner->plan));
66  } catch (std::bad_alloc &e) {
67  errno = ENOMEM;

69  // planner copy ctor can throw runtime_error, resulting in nullptr
70  if (np == nullptr)

75  try {
76  np = new planner_t ();

130 bool planner_multi::operator== (const planner_multi &o) const
131 {
132  if (m_span_counter != o.m_span_counter)
133  return false;
134  if (m_span_lookup != o.m_span_lookup)
135  return false;
136  if (m_iter.on_or_after != o.m_iter.on_or_after)
137  return false;
138  if (m_iter.duration != o.m_iter.duration)
139  return false;
140  if (m_iter.counts != o.m_iter.counts)
141  return false;
142 
143  if (m_types_totals_planners.size () != o.m_types_totals_planners.size ())
144  return false;
145  auto &o_by_type = o.m_types_totals_planners.get<res_type> ();
146  auto &by_type = m_types_totals_planners.get<res_type> ();
147  for (auto data : by_type) {
148  auto o_data = o_by_type.find (data.resource_type);
149  if (o_data == o_by_type.end ())
150  return false;
151  if (data.resource_type != o_data->resource_type)
152  return false;
153  if (data.resource_total != o_data->resource_total)
154  return false;
155  if (*(data.planner->plan) != *(o_data->planner->plan))
156  return false;
157  }
158 
159  return true;
160 }

185 void planner_multi::add_planner (int64_t base_time, uint64_t duration,
186  const uint64_t resource_total,
187  const char *resource_type, size_t i)
188 {
189  std::string type;
190  planner_t *p = nullptr;
191 
192  try {
193  type = std::string (resource_type);
194  p = new planner_t (base_time, duration,
195  resource_total,
196  resource_type);
197  } catch (std::bad_alloc &e) {
198  errno = ENOMEM;
199  throw std::bad_alloc ();
200  }
201  m_iter.counts[type] = 0;
202  if (i > m_types_totals_planners.size ())
203  m_types_totals_planners.push_back ({type, resource_total, p});
204  else {
205  auto it = m_types_totals_planners.begin () + i;
206  m_types_totals_planners.insert (
207  it, planner_multi_meta{type, resource_total, p});
208  }
209 
210 }

212 void planner_multi::delete_planners (const std::unordered_set<std::string> &rtypes)
213 {
214  auto &by_res = m_types_totals_planners.get<res_type> ();
215  for (auto iter = by_res.begin (); iter != by_res.end ();) {
216  if (rtypes.find (iter->resource_type) == rtypes.end ()) {
217  // need to remove from request_multi
218  m_iter.counts.erase (iter->resource_type);
219  // Trigger planner destructor
220  delete iter->planner;
221  iter = by_res.erase (iter);
222  } else
223  ++iter;
224  }
225 }
