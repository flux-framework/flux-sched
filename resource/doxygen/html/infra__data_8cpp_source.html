113 bool pool_infra_t::operator== (const pool_infra_t &o) const
114 {
115  if (tags != o.tags)
116  return false;
117  if (x_spans != o.x_spans)
118  return false;
119  if (job2span != o.job2span)
120  return false;
121  if (colors != o.colors)
122  return false;
123  if (!planners_equal (x_checker, o.x_checker))
124  return false;
125  if (subplans.size () != o.subplans.size ())
126  return false;
127  for (auto const &this_it : subplans) {
128  auto const other = o.subplans.find (this_it.first);
129  if (other == o.subplans.end ())
130  return false;
131  if (this_it.first != other->first)
132  return false;
133  if (!planner_multis_equal (this_it.second, other->second))
134  return false;
135  }
136 
137  return true;
138 }
