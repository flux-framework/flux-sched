73  template<class compare_op>
74  int choose_best_k (unsigned int k, compare_op comp)
75  {
76  if (k == 0 || k > m_qual_count) {
77  errno = EINVAL;
78  return -1;
79  }
80 
81  unsigned int i = 0;
82  int old = (int)m_best_k;
83  int to_be_selected = k;
84  std::sort (m_eval_egroups.begin (), m_eval_egroups.end (), comp);
85  while (to_be_selected > 0) {
86  if (to_be_selected <= (int)m_eval_egroups[i].count)
87  m_eval_egroups[i].needs = to_be_selected;
88  else
89  m_eval_egroups[i].needs = m_eval_egroups[i].count;
90 
91  to_be_selected -= m_eval_egroups[i].count;
92  i++;
93  }
94  m_best_k = k;
95  m_best_i = i;
96  return old;
97  }
98 
99  template<class binary_op>
100  int64_t accum_best_k (binary_op accum, int init=0)
101  {
102  if (m_best_k == 0 || m_best_i == 0) {
103  errno = EINVAL;
104  return -1;
105  }
106  int64_t score_accum = init;
107  for (unsigned int i = 0; i < m_best_i; i++)
108  score_accum = accum (score_accum, m_eval_egroups[i]);
109  return score_accum;
110  }
