164 double get_elapse_time (timeval &st, timeval &et)
165 {
166  double ts1 = (double)st.tv_sec + (double)st.tv_usec/1000000.0f;
167  double ts2 = (double)et.tv_sec + (double)et.tv_usec/1000000.0f;

171 static int run_match (std::shared_ptr<resource_context_t> &ctx, int64_t jobid,
172  const std::string cmd, const std::string &jobspec_fn,
173  Flux::Jobspec::Jobspec &job)
174 {
175  int rc = 0;
176  int rc2 = 0;
177  bool sat = true;
178  int64_t at = 0;
179  std::stringstream o;
180  double elapse = 0.0f;
181  unsigned int preorder_count = 0;
182  unsigned int postorder_count = 0;
183  struct timeval st, et;
184  std::ostream &out = (ctx->params.r_fname != "")? ctx->params.r_out
185  : std::cout;
186 
187  if ( (rc = gettimeofday (&st, NULL)) < 0) {
188  std::cerr << "ERROR: gettimeofday: " << strerror (errno) << std::endl;
189  goto done;
190  }
191 
192  if (cmd == "allocate")
193  rc2 = ctx->traverser->run (job, ctx->writers, match_op_t::
194  MATCH_ALLOCATE, (int64_t)jobid, &at);
195  else if (cmd == "allocate_with_satisfiability")
196  rc2 = ctx->traverser->run (job, ctx->writers, match_op_t::
197  MATCH_ALLOCATE_W_SATISFIABILITY,
198  (int64_t)jobid, &at);
199  else if (cmd == "allocate_orelse_reserve")
200  rc2 = ctx->traverser->run (job, ctx->writers, match_op_t::
201  MATCH_ALLOCATE_ORELSE_RESERVE,
202  (int64_t)jobid, &at);
203  else if (cmd == "satisfiability")
204  rc2 = ctx->traverser->run (job, ctx->writers, match_op_t::
205  MATCH_SATISFIABILITY,
206  (int64_t)jobid, &at);
207  else
208  goto done;
209 
210  if ((rc2 != 0) && (errno == ENODEV))
211  sat = false;
212 
213  if (ctx->traverser->err_message () != "") {
214  std::cerr << "ERROR: " << ctx->traverser->err_message ();
215  ctx->traverser->clear_err_message ();
216  }
217  if ( (rc = ctx->writers->emit (o)) < 0) {
218  std::cerr << "ERROR: match writer emit: "
219  << strerror (errno) << std::endl;
220  goto done;
221  }
222 
223  out << o.str ();
224 
225  if ( (rc = gettimeofday (&et, NULL)) < 0) {
226  std::cerr << "ERROR: gettimeofday: " << strerror (errno) << std::endl;
227  goto done;
228  }
229 
230  elapse = get_elapse_time (st, et);
231  preorder_count = ctx->traverser->get_total_preorder_count ();
232  postorder_count = ctx->traverser->get_total_postorder_count ();
233  update_match_perf (ctx, elapse);
234 
235  if (cmd != "satisfiability")
236  print_schedule_info (ctx, out, jobid,
237  jobspec_fn, rc2 == 0, at, sat,
238  elapse, preorder_count, postorder_count);
239  else
240  print_sat_info (ctx, out, sat, elapse, preorder_count, postorder_count);
241 
242 done:
243  return rc + rc2;
244 }

246 int cmd_match (std::shared_ptr<resource_context_t> &ctx,
247  std::vector<std::string> &args)
248 {
249  if (args.size () != 3) {
250  std::cerr << "ERROR: malformed command" << std::endl;
251  return 0;
252  }
253  std::string subcmd = args[1];
254  if (!(subcmd == "allocate" || subcmd == "allocate_orelse_reserve"
255  || subcmd == "allocate_with_satisfiability"
256  || subcmd == "satisfiability")) {
257  std::cerr << "ERROR: unknown subcmd " << args[1] << std::endl;
258  return 0;
259  }
260 
261  try {
262  int64_t jobid = ctx->jobid_counter;
263  std::string &jobspec_fn = args[2];
264  std::ifstream jobspec_in (jobspec_fn);
265  if (!jobspec_in) {
266  std::cerr << "ERROR: can't open " << jobspec_fn << std::endl;
267  return 0;
268  }
269  Flux::Jobspec::Jobspec job {jobspec_in};
270  jobspec_in.close ();
271 
272  run_match (ctx, jobid, subcmd, jobspec_fn, job);
273 
274  } catch (parse_error &e) {
275  std::cerr << "ERROR: Jobspec error for " << ctx->jobid_counter <<": "
276  << e.what () << std::endl;
277  }
278  return 0;

319 static int update_run (std::shared_ptr<resource_context_t> &ctx,
320  const std::string &fn, const std::string &str,
321  int64_t id, int64_t at, uint64_t d,
322  const std::string &reader)
323 {
324  int rc = -1;
325  double elapse = 0.0f;
326  std::stringstream o;
327  struct timeval st, et;
328  std::shared_ptr<resource_reader_base_t> rd;
329 
330  if (reader == "jgf") {
331  if ( (rd = create_resource_reader ("jgf")) == nullptr) {
332  std::cerr << "ERROR: can't create JGF reader " << std::endl;
333  return -1;
334  }
335  } else {
336  if ( (rd = create_resource_reader ("rv1exec")) == nullptr) {

338  return -1;

344  std::cerr << "ERROR: traverser run () returned error " << std::endl;
345  if (ctx->traverser->err_message () != "") {
346  std::cerr << "ERROR: " << ctx->traverser->err_message ();
347  ctx->traverser->clear_err_message ();
348  }
349  }
350  ctx->writers->emit (o);
351  std::ostream &out = (ctx->params.r_fname != "")? ctx->params.r_out
352  : std::cout;
353  out << o.str ();
354  gettimeofday (&et, NULL);
355 
356  elapse = get_elapse_time (st, et);
357  update_match_perf (ctx, elapse);
358  ctx->jobid_counter = id;
359  print_schedule_info (ctx, out, id, fn, rc == 0, at, true, elapse, 0, 0);
360 
361  return 0;
362 }

364 static int update (std::shared_ptr<resource_context_t> &ctx,
365  std::vector<std::string> &args)
366 {
367  uint64_t d = 0;
368  int64_t at = 0;
369  int64_t jobid = 0;
370  std::string subcmd = args[1];
371  std::string reader = args[2];
372  std::stringstream buffer{};
373 
374  if (!(subcmd == "allocate" || subcmd == "reserve")) {
375  std::cerr << "ERROR: unknown subcmd " << args[1] << std::endl;
376  return -1;
377  }
378  if (!(reader == "jgf" || reader == "rv1exec")) {
379  std::cerr << "ERROR: unsupported reader " << args[2] << std::endl;
380  return -1;
381  }
382  std::ifstream jgf_file (args[3]);
383  if (!jgf_file) {
384  std::cerr << "ERROR: can't open " << args[3] << std::endl;
385  return -1;
386  }
387 
388  jobid = static_cast<int64_t> (std::strtoll (args[4].c_str (), NULL, 10));
389  if (ctx->allocations.find (jobid) != ctx->allocations.end ()
390  || ctx->reservations.find (jobid) != ctx->reservations.end ()) {
391  std::cerr << "ERROR: existing Jobid " << std::endl;
392  return -1;
393  }
394  at = static_cast<int64_t> (std::strtoll (args[5].c_str (), NULL, 10));
395  d = static_cast<int64_t> (std::strtoll (args[6].c_str (), NULL, 10));
396  if (at < 0 || d == 0) {
397  std::cerr << "ERROR: invalid time ("
398  << at << ", " << d << ")" << std::endl;
399  return -1;
400  }
401 
402  buffer << jgf_file.rdbuf ();
403  jgf_file.close ();
404 
405  return update_run (ctx, args[3], buffer.str (), jobid, at, d, reader);
406 }

571 int cmd_cancel (std::shared_ptr<resource_context_t> &ctx,
572  std::vector<std::string> &args)
