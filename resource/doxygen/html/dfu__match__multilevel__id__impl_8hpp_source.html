36 template<typename FOLD>
37 int64_t multilevel_id_t<FOLD>::score_factor_t::calc_factor (
38  int64_t base_factor,
39  int64_t break_tie)
40 {
41  int64_t add, mul, tie;
42  if (base_factor < 0 || break_tie < 0) {
43  errno = EINVAL;
44  return -1;
45  }
46 
47  if (base_factor > (std::numeric_limits<int64_t>::max () - m_add_by)) {
48  errno = EOVERFLOW;
49  return -1;
50  }
51  add = base_factor + m_add_by;
52 
53  if (add > (std::numeric_limits<int64_t>::max () / m_multiply_by)) {
54  errno = EOVERFLOW;
55  return -1;
56  }
57  mul = add * m_multiply_by;
58  tie = abs (break_tie % static_cast<int64_t> (m_multiply_by) - 1);
59 
60  if (mul > (std::numeric_limits<int64_t>::max () - tie)) {
61  errno = EOVERFLOW;
62  return -1;
63  }
64  m_factor = mul + tie;
65  return m_factor;
66 }

114 template<typename FOLD>
115 int multilevel_id_t<FOLD>::dom_finish_graph (
116  const subsystem_t &subsystem,
117  const std::vector<
118  Flux::Jobspec::Resource> &resources,
119  const f_resource_graph_t &g,
120  scoring_api_t &dfu)
121 {
122  int64_t score = MATCH_MET;
123  for (auto &resource : resources) {
124  const std::string &type = resource.type;
125  unsigned qc = dfu.qualified_count (subsystem, type);
126  unsigned count = calc_count (resource, qc);
127  if (count == 0) {
128  score = MATCH_UNMET;
129  break;
130  }
131  dfu.choose_accum_best_k (subsystem, type, count, m_comp);
132  }
133  dfu.set_overall_score (score);
134  m_multilevel_scores = 0;
135  return (score == MATCH_MET)? 0 : -1;
136 }

149 template<typename FOLD>
150 int multilevel_id_t<FOLD>::dom_discover_vtx (
151  vtx_t u,
152  const subsystem_t &subsystem,
153  const std::vector<
154  Flux::Jobspec::Resource> &resources,
155  const f_resource_graph_t &g)
156 {
157  if (m_multilevel_factors.find (g[u].type) != m_multilevel_factors.end ()) {
158  if (g[u].id < -1)
159  return -1;
160  auto &f = m_multilevel_factors[g[u].type];
161  int64_t factor = f.calc_factor (g[u].id+1, g[u].uniq_id);
162  if (factor < 0)
163  return factor;
164  m_multilevel_scores += factor;
165  }
166  incr ();
167  return 0;
168 }

170 template<typename FOLD>
171 int multilevel_id_t<FOLD>::dom_finish_vtx (
172  vtx_t u,
173  const subsystem_t &subsystem,
174  const std::vector<
175  Flux::Jobspec::Resource> &resources,
176  const f_resource_graph_t &g,
177  scoring_api_t &dfu)
178 {
179  int64_t score = MATCH_MET;
180  int64_t overall;
181 
182  for (auto &resource : resources) {
183  if (resource.type != g[u].type)
184  continue;
185  // Jobspec resource type matches with the visiting vertex
186  for (auto &c_resource : resource.with) {
187  // Test children resource count requirements
188  const std::string &c_type = c_resource.type;
189  unsigned qc = dfu.qualified_count (subsystem, c_type);
190  unsigned count = calc_count (c_resource, qc);
191  if (count == 0) {
192  score = MATCH_UNMET;
193  break;
194  }
195  dfu.choose_accum_best_k (subsystem, c_type, count, m_comp);
196  }
197  }
198 
199  auto factor = m_multilevel_factors.find (g[u].type);
200  if (factor != m_multilevel_factors.end ())
201  m_multilevel_scores -= factor->second.m_factor;
202 
203  overall = (score == MATCH_MET)? (score + m_multilevel_scores
204  + g[u].id + 1) : score;
205  dfu.set_overall_score (overall);
206  decr ();
207  return (score == MATCH_MET)? 0 : -1;
208 }

225 template<typename FOLD>
226 int multilevel_id_t<FOLD>::add_score_factor (const std::string &type,
227  unsigned add_by,
228  unsigned multiple_by)
229 {
230  try {
231  if (m_multilevel_factors.find (type) != m_multilevel_factors.end ()) {
232  errno = EEXIST;
233  return -1;
234  }
235  auto ret = m_multilevel_factors.insert (
236  std::pair<std::string, score_factor_t> (
237  type, {type, add_by, multiple_by}));
238  if (!ret.second) {
239  errno = ENOMEM;
240  return -1;
241  }
242  return 0;
243  } catch (std::bad_alloc &e) {
244  errno = ENOMEM;
245  return -1;
246  }
