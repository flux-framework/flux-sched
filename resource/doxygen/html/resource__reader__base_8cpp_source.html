47 int resource_reader_base_t::set_allowlist (const std::string &csl)
48 {
49  if (csl == "")
50  return 0;
51 
52  int rc = -1;
53  size_t pos = 0;
54  std::string csl_copy = csl;
55  std::string sep = ",";
56 
57  try {
58  while ((pos = csl_copy.find (sep)) != std::string::npos) {
59  std::string resource = csl_copy.substr (0, pos);
60  if (resource != "")
61  allowlist.insert (resource);
62  csl_copy.erase (0, pos + sep.length ());
63  }
64  if (csl_copy != "")
65  allowlist.insert (csl_copy);
66  errno = EINVAL;
67  rc = allowlist.empty ()? -1 : 0;
68  } catch (std::out_of_range &e) {
69  errno = EINVAL;
70  rc = -1;
71  } catch (std::bad_alloc &e) {
72  errno = ENOMEM;
73  rc = -1;
74  }
75 
76  return rc;
77 }

94 int resource_reader_base_t::split_hostname (const std::string &hn,
95  std::string &basename,
96  int64_t &id) const
97 {
98  std::string suffix;
99  std::size_t first;
100  basename = hn;
101  std::size_t last = basename.find_last_not_of ("0123456789");
102 
103  if (last == (basename.size () - 1)) {
104  id = -1;
105  return 0;
106  }
107  if (last != std::string::npos) {
108  // has numeric suffix
109  suffix = basename.substr (last + 1);
110  basename = basename.substr (0, last + 1);
111  } else {
112  // all numbers
113  suffix = basename;
114  }
115 
116  first = suffix.find_first_not_of ("0");
117  if (first == std::string::npos) {
118  id = 0; // all 0s
119  return 0;
120  }
121 
122  suffix = suffix.substr (first);
123  try {
124  id = std::stoll (suffix);
125  } catch (std::invalid_argument &e) {
126  errno = EINVAL;
127  goto error;
128  } catch (std::out_of_range &e) {
129  errno = EOVERFLOW;
130  goto error;
131  }
132  return 0;
133 
134 error:
135  return -1;
136 }
