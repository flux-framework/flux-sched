38 size_t expr_eval_api_t::find_closing_paren (const std::string &e,
39  size_t at) const
40 {
41  int bal;
42  std::size_t pa;
43 
44  if ( (pa = e.find_first_not_of (" \t", at)) == std::string::npos) {
45  errno = EINVAL;
46  goto done;
47  }
48  bal = 1;
49  pa++;
50  while (bal && (pa = e.find_first_of ("()", pa)) != std::string::npos) {
51  bal = (e[pa] == ')')? bal - 1 : bal + 1;
52  pa++;
53  }
54 done:

96 expr_eval_api_t::pred_op_t expr_eval_api_t::parse_pred_op (const std::string &e,
97  size_t at,
98  size_t &next) const
99 {
100  std::size_t nws;
101  std::size_t xws;
102  pred_op_t op = pred_op_t::UNKNOWN;
103 
104  if (e[at] != ' ' && e[at] != '\t') {
105  errno = EINVAL;
106  goto done;
107  }
108  if ( (nws = e.find_first_not_of (" \t", at)) == std::string::npos) {
109  errno = EINVAL;
110  goto done;
111  }
112  if ( (xws = e.find_first_of (" \t", nws)) == std::string::npos)
113  xws = e.length ();
114  op = pred_op_t::AND;
115  next = at;
116  if (e.substr (nws, xws - nws) == "and") {
117  next = xws;
118  } else if (e.substr (nws, xws - nws) == "or") {
119  op = pred_op_t::OR;
120  next = xws;
121  }
122 
123 done:
124  return op;
