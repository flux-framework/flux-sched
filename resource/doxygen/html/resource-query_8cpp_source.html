55 static void usage (int code)
56 {
57  std::cerr << R"(
58 usage: resource-query [OPTIONS...]
59 
60 Command-line utility that takes in an HPC resource request written in
61 Flux's Canonical Job Specification (or simply a jobspec) (RFC 14) and
62 selects the best-matching compute and other resources in accordance
63 with a selection policy.
64 
65 Read in a resource-graph generation recipe written in the GRUG format
66 and populate the resource-graph data store representing the compute and
67 other HPC resources and their relationships (RFC 4).
68 
69 Provide a simple command-line interface (cli) to allow users to allocate
70 or reserve the resource set in this resource-graph data store
71 using a jobspec as an input.
72 Traverse the resource graph in a predefined order for resource selection.
73 Currently only support one traversal type: depth-first traversal on the
74 dominant subsystem and up-walk traversal on one or more auxiliary
75 subsystems.
76 
77 OPTIONS allow for using a predefined matcher that is configured
78 to use a different set of subsystems as its dominant and/or auxiliary
79 ones to perform the matches on.
80 
81 OPTIONS also allow for instantiating a different resource-matching
82 selection policy--e.g., select resources with high or low IDs first.
83 
84 OPTIONS allow for exporting the filtered graph of the used matcher
85 in a selected graph format at the end of the cli session.
86 
87 To see cli commands, type in "help" in the cli: i.e.,
88  % resource-query> help
89 
90 OPTIONS:
91  -h, --help
92  Display this usage information
93 
94  -L, --load-file=filepath
95  Input file from which to load the resource graph data store
96  (default=conf/default)
97 
98  -f, --load-format=<grug|hwloc|jgf|rv1exec>
99  Format of the load file (default=grug)
100 
101  -W, --load-allowlist=<resource1[,resource2[,resource3...]]>
102  Allowlist of resource types to be loaded
103  Resources that are not included in this list will be filtered out
104 
105  -S, --match-subsystems="
106  "<CA|IBA|IBBA|PFS1BA|PA|C+IBA|C+PFS1BA|C+PA|IB+IBBA|"
107  "C+P+IBA|VA|V+PFS1BA|ALL>
108  Set the predefined matcher to use. Available matchers are:
109  CA: Containment Aware
110  IBA: InfiniBand connection-Aware
111  IBBA: InfiniBand Bandwidth-Aware
112  PFS1BA: Parallel File System 1 Bandwidth-aware
113  PA: Power-Aware
114  C+IBA: Containment- and InfiniBand connection-Aware
115  C+PFS1BA: Containment- and PFS1 Bandwidth-Aware
116  C+PA: Containment- and Power-Aware
117  IB+IBBA: InfiniBand connection and Bandwidth-Aware
118  C+P+IBA: Containment-, Power- and InfiniBand connection-Aware
119  VA: Virtual Hierarchy-Aware
120  V+PFS1BA: Virtual Hierarchy and PFS1 Bandwidth-Aware
121  ALL: Aware of everything.
122  (default=CA).
123 
124  -P, --match-policy=<low|high|lonode|hinode|lonodex|hinodex|first|firstnodex|locality|variation>
125  Set the resource match selection policy. Available policies are:
126  low: Select resources with low ID first
127  high: Select resources with high ID first
128  lonode: Select resources with lowest node ID first,

242 static int set_subsystems_use (std::shared_ptr<resource_context_t> &ctx,
243  std::string n)
244 {
245  int rc = 0;
246  ctx->matcher->set_matcher_name (n);
247  dfu_match_cb_t &matcher = *(ctx->matcher);
248  const std::string &matcher_type = matcher.matcher_name ();
249 
250  if (boost::iequals (matcher_type, std::string ("CA"))) {
251  if ( (rc = subsystem_exist (ctx, "containment")) == 0)
252  matcher.add_subsystem ("containment", "*");
253  } else if (boost::iequals (matcher_type, std::string ("IBA"))) {
254  if ( (rc = subsystem_exist (ctx, "ibnet")) == 0)
255  matcher.add_subsystem ("ibnet", "*");
256  } else if (boost::iequals (matcher_type, std::string ("IBBA"))) {
257  if ( (rc = subsystem_exist (ctx, "ibnetbw")) == 0)
258  matcher.add_subsystem ("ibnetbw", "*");
259  } else if (boost::iequals (matcher_type, std::string ("PFS1BA"))) {
260  if ( (rc = subsystem_exist (ctx, "pfs1bw")) == 0)
261  matcher.add_subsystem ("pfs1bw", "*");
262  } else if (boost::iequals (matcher_type, std::string ("PA"))) {
263  if ( (rc = subsystem_exist (ctx, "power")) == 0)
264  matcher.add_subsystem ("power", "*");
265  } else if (boost::iequals (matcher_type, std::string ("C+PFS1BA"))) {
266  if ( (rc = subsystem_exist (ctx, "containment")) == 0)
267  matcher.add_subsystem ("containment", "contains");
268  if ( !rc && (rc = subsystem_exist (ctx, "pfs1bw")) == 0)
269  matcher.add_subsystem ("pfs1bw", "*");
270  } else if (boost::iequals (matcher_type, std::string ("C+IBA"))) {
271  if ( (rc = subsystem_exist (ctx, "containment")) == 0)
272  matcher.add_subsystem ("containment", "contains");
273  if ( !rc && (rc = subsystem_exist (ctx, "ibnet")) == 0)
274  matcher.add_subsystem ("ibnet", "connected_up");
275  } else if (boost::iequals (matcher_type, std::string ("C+PA"))) {
276  if ( (rc = subsystem_exist (ctx, "containment")) == 0)
277  matcher.add_subsystem ("containment", "*");
278  if ( !rc && (rc = subsystem_exist (ctx, "power")) == 0)
279  matcher.add_subsystem ("power", "draws_from");
280  } else if (boost::iequals (matcher_type, std::string ("IB+IBBA"))) {
281  if ( (rc = subsystem_exist (ctx, "ibnet")) == 0)
282  matcher.add_subsystem ("ibnet", "connected_down");
283  if ( !rc && (rc = subsystem_exist (ctx, "ibnetbw")) == 0)
284  matcher.add_subsystem ("ibnetbw", "*");
285  } else if (boost::iequals (matcher_type, std::string ("C+P+IBA"))) {
286  if ( (rc = subsystem_exist (ctx, "containment")) == 0)
287  matcher.add_subsystem ("containment", "contains");
288  if ( (rc = subsystem_exist (ctx, "power")) == 0)
289  matcher.add_subsystem ("power", "draws_from");
290  if ( !rc && (rc = subsystem_exist (ctx, "ibnet")) == 0)
291  matcher.add_subsystem ("ibnet", "connected_up");
292  } else if (boost::iequals (matcher_type, std::string ("V+PFS1BA"))) {
293  if ( (rc = subsystem_exist (ctx, "virtual1")) == 0)
294  matcher.add_subsystem ("virtual1", "*");
295  if ( !rc && (rc = subsystem_exist (ctx, "pfs1bw")) == 0)
296  matcher.add_subsystem ("pfs1bw", "*");
297  } else if (boost::iequals (matcher_type, std::string ("VA"))) {
298  if ( (rc = subsystem_exist (ctx, "virtual1")) == 0)
299  matcher.add_subsystem ("virtual1", "*");
300  } else if (boost::iequals (matcher_type, std::string ("ALL"))) {
301  if ( (rc = subsystem_exist (ctx, "containment")) == 0)
302  matcher.add_subsystem ("containment", "*");
303  if ( !rc && (rc = subsystem_exist (ctx, "ibnet")) == 0)
304  matcher.add_subsystem ("ibnet", "*");
305  if ( !rc && (rc = subsystem_exist (ctx, "ibnetbw")) == 0)
306  matcher.add_subsystem ("ibnetbw", "*");
307  if ( !rc && (rc = subsystem_exist (ctx, "pfs1bw")) == 0)
308  matcher.add_subsystem ("pfs1bw", "*");
309  if ( (rc = subsystem_exist (ctx, "power")) == 0)
310  matcher.add_subsystem ("power", "*");
311  } else {
312  rc = -1;
313  }
314  return rc;
315 }

327 static void flatten (f_resource_graph_t &fg,
328  std::map<vtx_t, std::string> &paths,
329  std::map<vtx_t, std::string> &subsystems,
330  std::map<edg_t, std::string> &esubsystems,
331  std::map<vtx_t, std::string> &properties)
332 {
333  f_vtx_iterator_t vi, v_end;
334  f_edg_iterator_t ei, e_end;
335 
336  for (tie (vi, v_end) = vertices (fg); vi != v_end; ++vi) {
337  paths[*vi] = "{";
338  for (auto &kv : fg[*vi].paths) {
339  if (paths[*vi].size () > 1)
340  paths[*vi] += ",";
341  paths[*vi] += kv.first + "=" + kv.second;
342  }
343  paths[*vi] += "}";
344  subsystems[*vi] = "{";
345  for (auto &kv : fg[*vi].idata.member_of) {
346  if (subsystems[*vi].size () > 1)
347  subsystems[*vi] += ",";
348  subsystems[*vi] += kv.first + "=" + kv.second;
349  }
350  subsystems[*vi] += "}";
351  properties[*vi] = "{";
352  for (auto &kv : fg[*vi].properties) {
353  if (properties[*vi].size () > 1)
354  properties[*vi] += ",";
355  properties[*vi] += kv.first + "=" + kv.second;
356  }
357  properties[*vi] += "}";
358  }
359  for (tie (ei, e_end) = edges (fg); ei != e_end; ++ei) {
360  esubsystems[*ei] = "{";
361  for (auto &kv : fg[*ei].idata.member_of) {
362  if (esubsystems[*ei].size () > 0)
363  esubsystems[*ei] += ",";
364  esubsystems[*ei] += kv.first + "=" + kv.second;
365  }
366  esubsystems[*ei] += "}";
367  }
368 }

370 static void write_to_graphml (f_resource_graph_t &fg, std::fstream &o)
371 {
372  boost::dynamic_properties dp;
373  std::map<edg_t, std::string> esubsystems;
374  std::map<vtx_t, std::string> subsystems, properties, paths;
375  boost::associative_property_map<
376  std::map<vtx_t, std::string>> subsystems_map (subsystems);
377  boost::associative_property_map<
378  std::map<edg_t, std::string>> esubsystems_map (esubsystems);
379  boost::associative_property_map<
380  std::map<vtx_t, std::string>> props_map (properties);
381  boost::associative_property_map<
382  std::map<vtx_t, std::string>> paths_map (paths);
383 
384  flatten (fg, paths, subsystems, esubsystems, properties);
385 
386  // Resource pool vertices
387  dp.property ("paths", paths_map);
388  dp.property ("props", props_map);
389  dp.property ("member_of", subsystems_map);
390  dp.property ("type", get (&resource_pool_t::type, fg));
391  dp.property ("basename", get (&resource_pool_t::basename, fg));
392  dp.property ("name", get (&resource_pool_t::name, fg));
393  dp.property ("id", get (&resource_pool_t::id, fg));
394  dp.property ("uniq_id", get (&resource_pool_t::uniq_id, fg));
395  dp.property ("size", get (&resource_pool_t::size, fg));
396  dp.property ("unit", get (&resource_pool_t::unit, fg));
397  // Relation edges
398  dp.property ("member_of", esubsystems_map);
399 
400  write_graphml (o, fg, dp, true);
401 }

432 static void control_loop (std::shared_ptr<resource_context_t> &ctx)
433 {
434  cmd_func_f *cmd = NULL;
435  while (1) {
436  char *line = ctx->params.disable_prompt? readline ("")
437  : readline ("resource-query> ");
438  if (line == NULL)
439  continue;
440  else if (*line)
441  add_history (line);
442 
443  std::vector<std::string> tokens;
444  std::istringstream iss (line);
445  std::copy (std::istream_iterator<std::string> (iss),
446  std::istream_iterator<std::string> (),
447  back_inserter (tokens));
448  free(line);
449  if (tokens.empty ())
450  continue;
451 
452  std::string &cmd_str = tokens[0];
453  if (!(cmd = find_cmd (cmd_str)))
454  continue;
455  if (cmd (ctx, tokens) != 0)
456  break;
457  }
458 }

460 static int populate_resource_db (std::shared_ptr<resource_context_t> &ctx)
461 {
462  int rc = -1;
463  double elapse;
464  std::ifstream in_file;
465  struct timeval st, et;
466  std::stringstream buffer{};
467  std::shared_ptr<resource_reader_base_t> rd;
468 
469  if (ctx->params.reserve_vtx_vec != 0)
470  ctx->db->resource_graph.m_vertices.reserve (
471  ctx->params.reserve_vtx_vec);
472  if ( (rd = create_resource_reader (ctx->params.load_format)) == nullptr) {
473  std::cerr << "ERROR: Can't create load reader " << std::endl;
474  goto done;
475  }
476  if (ctx->params.load_allowlist != "") {
477  if (rd->set_allowlist (ctx->params.load_allowlist) < 0)
478  std::cerr << "ERROR: Can't set allowlist" << std::endl;
479  if (!rd->is_allowlist_supported ())
480  std::cout << "WARN: allowlist unsupported" << std::endl;
481  }
482 
483  in_file.open (ctx->params.load_file.c_str (), std::ifstream::in);
484  if (!in_file.good ()) {
485  std::cerr << "ERROR: Can't open " << ctx->params.load_file << std::endl;
486  goto done;
487  }
488  buffer << in_file.rdbuf ();
489  in_file.close ();
490 
491  gettimeofday (&st, NULL);
492  if ( (rc = ctx->db->load (buffer.str (), rd)) != 0) {
493  std::cerr << "ERROR: " << rd->err_message () << std::endl;
494  std::cerr << "ERROR: error in generating resources" << std::endl;
495  goto done;
496  }
497  gettimeofday (&et, NULL);
498 
499  elapse = get_elapse_time (st, et);
500  if (ctx->params.elapse_time) {
501  resource_graph_t &g = ctx->db->resource_graph;
502  std::cout << "INFO: Graph Load Time: " << elapse << std::endl;
503  std::cout << "INFO: Vertex Count: " << num_vertices (g) << std::endl;
504  std::cout << "INFO: Edge Count: " << num_edges (g) << std::endl;
505  std::cout << "INFO: by_type Key-Value Pairs: " <<
506  ctx->db->metadata.by_type.size () << std::endl;
507  std::cout << "INFO: by_name Key-Value Pairs: " <<
508  ctx->db->metadata.by_name.size () << std::endl;
509  std::cout << "INFO: by_path Key-Value Pairs: " <<
510  ctx->db->metadata.by_path.size () << std::endl;
511  for (auto it = ctx->db->metadata.by_rank.begin ();
512  it != ctx->db->metadata.by_rank.end (); ++it) {
513  std::cout << "INFO: number of vertices with rank "
514  << it->first << ": " << it->second.size () << "\n";
515  }
516  }
517 
518 done:
519  return rc;
520 }

549 static int init_resource_graph (std::shared_ptr<resource_context_t> &ctx)
550 {
551  int rc = 0;
552 
553  if ( (rc = populate_resource_db (ctx)) != 0) {

611 static void process_args (std::shared_ptr<resource_context_t> &ctx,
612  int argc, char *argv[])
613 {
614  int rc = 0;
615  int ch = 0;
616  std::string token;
617 
618  while ((ch = getopt_long (argc, argv, OPTIONS, longopts, NULL)) != -1) {
619  switch (ch) {
620  case 'h': /* --help */
621  usage (0);
622  break;
623  case 'L': /* --load-file */
624  ctx->params.load_file = optarg;
625  if (!fs::exists(ctx->params.load_file)) {
626  std::cerr << "[ERROR] file does not exist for --load-file: ";
627  std::cerr << optarg << std::endl;
628  usage (1);
629  } else if (fs::is_directory(ctx->params.load_file)) {
630  std::cerr << "[ERROR] path passed to --load-file is a directory: ";
631  std::cerr << optarg << std::endl;
632  usage (1);
633  }
634  break;
635  case 'f': /* --load-format */
636  ctx->params.load_format = optarg;
637  if (!known_resource_reader (ctx->params.load_format)) {
638  std::cerr << "[ERROR] unknown format for --load-format: ";
639  std::cerr << optarg << std::endl;
640  usage (1);
641  }
642  break;
643  case 'W': /* --hwloc-allowlist */
644  token = optarg;
645  if(token.find_first_not_of(' ') != std::string::npos) {
646  ctx->params.load_allowlist += "cluster,";
647  ctx->params.load_allowlist += token;
648  }
649  break;
650  case 'S': /* --match-subsystems */
651  ctx->params.matcher_name = optarg;
652  break;
653  case 'P': /* --match-policy */
654  ctx->params.matcher_policy = optarg;
655  break;
656  case 'F': /* --match-format */
657  ctx->params.match_format = optarg;
658  if (!known_match_format (ctx->params.match_format)) {
659  std::cerr << "[ERROR] unknown format for --match-format: ";
660  std::cerr << optarg << std::endl;
661  usage (1);
662  }
663  break;
664  case 'g': /* --graph-format */
665  rc = string_to_graph_format (optarg, ctx->params.o_format);
666  if ( rc != 0) {

668  std::cerr << optarg << std::endl;
669  usage (1);
670  }
671  graph_format_to_ext (ctx->params.o_format, ctx->params.o_fext);
672  break;
673  case 'o': /* --graph-output */
674  ctx->params.o_fname = optarg;
675  break;
676  case 'p': /* --prune-filters */
677  token = optarg;
678  if(token.find_first_not_of(' ') != std::string::npos) {
679  ctx->params.prune_filters += ",";
680  ctx->params.prune_filters += token;
681  }
682  break;
683  case 't': /* --test-output */
684  ctx->params.r_fname = optarg;
685  break;
686  case 'r': /* --reserve-vtx-vec */
687  // If atoi fails, it defaults to 0, which is fine for us
688  ctx->params.reserve_vtx_vec = atoi (optarg);
689  if ( (ctx->params.reserve_vtx_vec < 0)

694  std::cerr << optarg << std::endl;
695  }
696  break;
697  case 'e': /* --elapse-time */
698  ctx->params.elapse_time = true;
699  break;
700  case 'd': /* --disable-prompt */
701  ctx->params.disable_prompt = true;
702  break;
703  default:
704  usage (1);
705  break;
706  }
707  }
708 
709  if (optind != argc)
710  usage (1);
