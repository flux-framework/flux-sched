61 std::unique_ptr<Constraint>
62 Flux::Jobspec::constraint_parser (const YAML::Node &constraint)
63 {
64  YAML::const_iterator it;
65  YAML::Node operands;
66 
67  if (!constraint.IsMap ())
68  throw parse_error (constraint, "constraint is not a mapping");
69  if (constraint.size () > 1)
70  throw parse_error (constraint,
71  "constraint map may not contain > 1 operation");
72  if (constraint.size () == 0)
73  return std::unique_ptr<Constraint>(new Constraint());
74 
75  it = constraint.begin();
76  std::string operation = it->first.as<std::string>();
77  operands = it->second;
78 
79  if (!operands.IsSequence ()) {
80  std::string msg = operation + " operator value must be an array";
81  throw parse_error (operands, msg.c_str());
82  }
83 
84  if (operation == "properties")
85  return std::unique_ptr<PropertyConstraint>
86  (new PropertyConstraint (operands));
87  else if (operation == "hostlist")
88  return std::unique_ptr<HostlistConstraint>
89  (new HostlistConstraint (operands));
90  else if (operation == "ranks")
91  return std::unique_ptr<RankConstraint>
92  (new RankConstraint (operands));
93  else if (operation == "and" || operation == "or" || operation == "not")
94  return std::unique_ptr<ConditionalConstraint>
95  (new ConditionalConstraint (operation, operands));
96 
97  std::string msg = "unknown constraint operator: " + operation;
98  throw parse_error (constraint, msg.c_str());
99 }

119 PropertyConstraint::PropertyConstraint (const YAML::Node &properties)
120 {
121  for (auto&& property: properties) {
122  // YAML::Node::as<std:string> will work on a non-string YAML node.
123  // use Tag() instead to determine if the scalar is a quoted string,
124  // in which case "!" is returned.
125  //
126  // Ref: https://yaml.org/spec/1.2-old/spec.html#id2804923
127  if (!property.IsScalar() || property.Tag() != "!")
128  throw parse_error (properties, "non-string property specified");
129 
130  std::string prop = property.as<std::string>();
131  if (prop[0] == '^')
132  prop = prop.substr (1);
133  if (prop.find_first_of ("!&\'\"^`|()") != std::string::npos) {
134  std::string errmsg = property.as<std::string>() + " is invalid";
135  throw parse_error (properties, errmsg.c_str ());
136  }
137  values.push_back(property.as<std::string>());
138  }
139 }

141 bool PropertyConstraint::match (const Flux::resource_model::resource_t &r) const
142 {
143  for (auto&& value: values) {
144  std::string property = value;
145  bool negate = false;
146 
147  if (property[0] == '^') {
148  negate = true;
149  property = property.substr (1);
150  }
151  bool found = r.properties.find (property) != r.properties.end ();
152  if (negate)
153  found = !found;
154  if (!found)
155  return false;
156  }
157  return true;
158 }
