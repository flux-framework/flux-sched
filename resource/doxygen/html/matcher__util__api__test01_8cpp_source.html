63 int test_jobspec_template::emit (json_t *count_obj)
64 {
65  json_t *obj{nullptr};
66  json_t *tasks_obj{nullptr};
67  char *json_str{nullptr};
68 
69  if ( (tasks_obj = json_pack ("[{s:[s] s:s s:{s:i}}]",
70  "command", "app",
71  "slot", "foo",
72  "count",
73  "per_slot", 1)) == nullptr) {
74  errno = ENOMEM;
75  return -1;
76  }
77 
78  if ( (obj = json_pack (
79  "{s:i s:[{s:s s:s s:o s:[{s:s s:i}]}] s:o s:{}}",
80  "version", 9999,
81  "resources",
82  "type", "slot",
83  "label", "foo",
84  "count", count_obj,
85  "with",
86  "type", "node",
87  "count", 1,
88  "tasks", tasks_obj,
89  "attributes")) == nullptr) {
90  errno = ENOMEM;
91  return -1;
92  }
93 
94  if ( (json_str = json_dumps (obj, JSON_INDENT (0))) == nullptr) {
95  errno = ENOMEM;
96  return -1;
97  }
98 
99  m_job_spec_string = json_str;
100  json_decref (obj);
101  free (json_str);
102  return 0;
103 }
