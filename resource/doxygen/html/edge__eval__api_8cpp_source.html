176 int evals_t::merge (evals_t &o)
177 {
178  if (m_cutline != o.m_cutline || m_resrc_type != o.m_resrc_type) {
179  errno = EINVAL;
180  return -1;
181  }
182  m_qual_count += o.m_qual_count;
183  m_total_count += o.m_total_count;
184  m_cutline = o.m_cutline;
185  m_eval_egroups.insert (m_eval_egroups.end (), o.m_eval_egroups.begin (),
186  o.m_eval_egroups.end ());
187  return 0;
188 }
