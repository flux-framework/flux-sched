246 static void print_schedule_info (resource_query_t &ctx,
247  std::ostream &out, uint64_t jobid,
248  const std::string &jobspec_fn, bool matched,
249  int64_t at, bool sat, double elapse,
250  bool print_elapse)
251 {
252  unsigned int pre = reapi_cli_t::preorder_count (&ctx);

254  if (matched) {

258  std::string scheduled_at = (at == 0)? "Now" : std::to_string (at);
259  out << "INFO:" << " =============================" << std::endl;
260  out << "INFO:" << " JOBID=" << jobid << std::endl;
261  out << "INFO:" << " RESOURCES=" << mode << std::endl;
262  out << "INFO:" << " SCHEDULED AT=" << scheduled_at << std::endl;
263  if (print_elapse) {
264  std::cout << "INFO:" << " ELAPSE=" << std::to_string (elapse)
265  << std::endl;
266  std::cout << "INFO:" << " PREORDER VISIT COUNT=" << pre
267  << std::endl;
268  std::cout << "INFO:" << " POSTORDER VISIT COUNT=" << post
269  << std::endl;
270  }
271  out << "INFO:" << " =============================" << std::endl;
272  st = (at == 0)? job_lifecycle_t::ALLOCATED : job_lifecycle_t::RESERVED;
273  if (reapi_cli_t::info (&ctx, jobid, job) == 0)
274  job->jobspec_fn = jobspec_fn;
275  } else {

309 int match (resource_query_t &ctx, std::vector<std::string> &args,
310  json_t *params, std::ostream &out)
311 {
312  int rc = 0;
313  int64_t at = 0;
314  match_op_t match_op;
315  bool reserved = false;
316  bool sat = true;
317  bool matched = true;
318  std::string R = "";
319  double ov = 0.0;
320  std::stringstream o;
321  struct timeval st, et;
322 
323  if ( (gettimeofday (&st, NULL)) < 0) {
324  std::cerr << "ERROR: gettimeofday: " << strerror (errno) << std::endl;
325  return 0;
326  }
327 
328  if (args.size () != 3) {
329  std::cerr << "ERROR: malformed command" << std::endl;
330  return 0;
331  }
332  std::string subcmd = args[1];
333  if (subcmd == "allocate") {
334  match_op = match_op_t::MATCH_ALLOCATE;
335  } else if (subcmd == "allocate_orelse_reserve") {
336  match_op = match_op_t::MATCH_ALLOCATE_ORELSE_RESERVE;
337  } else if (subcmd == "allocate_with_satisfiability") {
338  match_op = match_op_t::MATCH_ALLOCATE_W_SATISFIABILITY;
339  } else if (subcmd == "satisfiability") {
340  match_op = match_op_t::MATCH_SATISFIABILITY;
341  } else {
342  std::cerr << "ERROR: unknown subcmd " << args[1] << std::endl;
343  return 0;
344  }
345 
346  uint64_t jobid = ctx.get_job_counter ();
347  std::string &jobspec_fn = args[2];
348  std::ifstream jobspec_in (jobspec_fn);
349  if (!jobspec_in) {
350  std::cerr << "ERROR: can't open " << jobspec_fn << std::endl;
351  return 0;
352  }
353  std::string jobspec ( (std::istreambuf_iterator<char> (jobspec_in) ),
354  (std::istreambuf_iterator<char> () ) );
355 
356  rc = reapi_cli_t::match_allocate (&ctx, match_op, jobspec, jobid, reserved, R, at, ov);
357 
358  // check for match success
359  if ( (errno == ENODEV) || (errno == EBUSY) || (errno == EINVAL))
360  matched = false;
361 
362  // check for satisfiability
363  if (errno == ENODEV)
364  sat = false;
365 
366  if (reapi_cli_t::get_err_message () != "") {
367  std::cerr << reapi_cli_t::get_err_message () << std::endl;
368  reapi_cli_t::clear_err_message ();
369  }
370 
371  out << R;
372 
373  if ( (gettimeofday (&et, NULL)) < 0) {
374  std::cerr << "ERROR: gettimeofday: " << strerror (errno) << std::endl;
375  return 0;
376  }
377 
378  if (subcmd != "satisfiability")
379  print_schedule_info (ctx, out, jobid,
380  jobspec_fn, matched, at, sat,
381  ov, json_object_get (params, "elapse_time"));
382  else
383  print_sat_info (ctx, out, sat, ov,
384  json_object_get (params, "elapse_time"));
385 
386  jobspec_in.close ();
387 
388  return 0;
389 }

434 static void process_args (json_t *options,
435  int argc, char *argv[])
436 {
437  int rc = 0;
438  int ch = 0;
439  std::string token;
440  Flux::resource_model::emit_format_t format;
441 
442 
443  /* set defaults specific for resource query */
444  json_object_set_new (options, "match_format", json_string ("simple"));
445 
446  while ( (ch = getopt_long (argc, argv, OPTIONS, longopts, NULL)) != -1) {
447  switch (ch) {
448  case 'h': /* --help */
449  usage (0);
450  break;
451  case 'L': /* --load-file */
452  json_object_set_new (options, "load_file", json_string (optarg));
453  if (!fs::exists (optarg)) {
454  std::cerr << "[ERROR] file does not exist for --load-file: ";
455  std::cerr << optarg << std::endl;
456  usage (1);
457  } else if (fs::is_directory (optarg)) {
458  std::cerr << "[ERROR] path passed to --load-file is a directory: ";
459  std::cerr << optarg << std::endl;
460  usage (1);
461  }
462  break;
463  case 'f': /* --load-format */
464  json_object_set_new (options, "load_format", json_string (optarg));
465  if (!known_resource_reader (optarg)) {
466  std::cerr << "[ERROR] unknown format for --load-format: ";
467  std::cerr << optarg << std::endl;
468  usage (1);
469  }
470  break;
471  case 'W': /* --hwloc-allowlist */
472  token = optarg;
473  if (token.find_first_not_of (' ') != std::string::npos) {
474  if (!json_object_get (options,"load_allowlist")) {
475  json_object_set_new (options, "load_allowlist", json_string (""));
476  }
477  json_object_set_new (options, "load_allowlist",
478  json_pack ("s++",
479  json_object_get (options,"load_allowlist"),
480  "cluster,", token.c_str ()));
481  }
482  break;
483  case 'S': /* --match-subsystems */
484  json_object_set_new (options, "matcher_name", json_string (optarg));
485  break;
486  case 'P': /* --match-policy */
487  json_object_set_new (options, "matcher_policy", json_string (optarg));
488  break;
489  case 'F': /* --match-format */
490  json_object_set_new (options, "match_format", json_string (optarg));
491  if (!known_match_format (optarg)) {
492  std::cerr << "[ERROR] unknown format for --match-format: ";
493  std::cerr << optarg << std::endl;
494  usage (1);
495  }
496  break;
497  case 'g': /* --graph-format */
498  rc = string_to_graph_format (optarg, format);
499  if ( rc != 0) {
500  std::cerr << "[ERROR] unknown format for --graph-format: ";
501  std::cerr << optarg << std::endl;
502  usage (1);
503  }
504  graph_format_to_ext (format, token);
505  json_object_set_new (options, "o_fext", json_string (token.c_str ()));
506  break;
507  case 'o': /* --graph-output */
508  json_object_set_new (options, "o_fname", json_string (optarg));
509  break;
510  case 'p': /* --prune-filters */
511  token = optarg;
512  if (token.find_first_not_of (' ') != std::string::npos) {
513  if (!json_object_get (options,"prune_filters")) {
514  json_object_set_new (options, "prune_filters", json_string (token.c_str ()));
515  } else {
516  json_object_set_new (options, "prune_filters",
517  json_pack ("s++",
518  json_object_get (options,"prune_filters"),
519  ",", token.c_str ()));
520  }
521 
522  }
523  break;
524  case 't': /* --test-output */
525  json_object_set_new (options, "r_fname", json_string (optarg));
526  break;
527  case 'r': /* --reserve-vtx-vec */
528  // If atoi fails, it defaults to 0, which is fine for us
529  json_object_set_new (options, "match_format", json_integer (atoi (optarg)));
530  if ( (atoi (optarg) < 0)
531  || (atoi (optarg) > 2000000)) {
532  json_object_set_new (options, "match_format", json_integer (0));
533  std::cerr
534  << "WARN: out of range specified for --reserve-vtx-vec: ";
535  std::cerr << optarg << std::endl;
536  }
537  break;
538  case 'e': /* --elapse-time */
539  json_object_set_new (options, "elapse_time", json_true ());
540  break;
541  case 'd': /* --disable-prompt */
542  json_object_set_new (options, "disable_prompt", json_true ());
543  break;
544  default:
545  usage (1);
546  break;
547  }
548  }
549 
550  if (optind != argc)
551  usage (1);
552 }

566 static void control_loop (resource_query_t &ctx, json_t *params,
567  std::ostream &out)
568 {
569  cmd_func_f *cmd = NULL;
