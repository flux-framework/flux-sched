62 int dfu_traverser_t::schedule (Jobspec::Jobspec &jobspec,
63  detail::jobmeta_t &meta, bool x, match_op_t op,
64  vtx_t root,
65  std::unordered_map<std::string, int64_t> &dfv)
66 {
67  int64_t t = 0;
68  int rc = -1;
69  size_t len = 0;
70  std::vector<uint64_t> agg;
71  uint64_t duration = 0;
72  int saved_errno = errno;
73  planner_multi_t *p = NULL;
74  const subsystem_t &dom = get_match_cb ()->dom_subsystem ();
75 
76  if ((rc = detail::dfu_impl_t::select (jobspec, root, meta, x)) == 0) {
77  m_total_preorder = detail::dfu_impl_t::get_preorder_count ();
78  m_total_postorder = detail::dfu_impl_t::get_postorder_count ();
79  goto out;
80  }
81 
82  /* Currently no resources/devices available... Do more... */
83  switch (op) {
84  case match_op_t::MATCH_ALLOCATE_W_SATISFIABILITY: {
85  /* With satisfiability check */
86  errno = EBUSY;
87  meta.alloc_type = jobmeta_t::alloc_type_t::AT_SATISFIABILITY;
88  p = (*get_graph ())[root].idata.subplans.at (dom);
89  meta.at = planner_multi_base_time (p)
90  + planner_multi_duration (p) - meta.duration - 1;
91  detail::dfu_impl_t::count_relevant_types (p, dfv, agg);
92  if (detail::dfu_impl_t::select (jobspec, root, meta, x) < 0) {
93  errno = (errno == EBUSY)? ENODEV : errno;
94  detail::dfu_impl_t::update ();
95  }
96  m_total_preorder += detail::dfu_impl_t::get_preorder_count ();
97  m_total_postorder += detail::dfu_impl_t::get_postorder_count ();
98  break;
99  }
100  case match_op_t::MATCH_ALLOCATE_ORELSE_RESERVE: {
101  /* Or else reserve */
102  errno = 0;
103  meta.alloc_type = jobmeta_t::alloc_type_t::AT_ALLOC_ORELSE_RESERVE;
104  t = meta.at + 1;
105  p = (*get_graph ())[root].idata.subplans.at (dom);
106  len = planner_multi_resources_len (p);
107  duration = meta.duration;
108  detail::dfu_impl_t::count_relevant_types (p, dfv, agg);
109  for (t = planner_multi_avail_time_first (p, t, duration, &agg[0], len);
110  (t != -1 && rc && !errno); t = planner_multi_avail_time_next (p)) {
111  meta.at = t;
112  rc = detail::dfu_impl_t::select (jobspec, root, meta, x);
113  m_total_preorder += detail::dfu_impl_t::get_preorder_count ();
114  m_total_postorder += detail::dfu_impl_t::get_postorder_count ();
115  }
116  // The planner layer returns
117  // ENOENT when no scheduleable point exists
118  // ERANGE when the total available core count at the root < the request
119  if (rc < 0 && (errno == ENOENT || errno == ERANGE)) {
120  errno = EBUSY;
121  meta.alloc_type = jobmeta_t::alloc_type_t::AT_SATISFIABILITY;
122  meta.at = planner_multi_base_time (p)
123  + planner_multi_duration (p) - duration - 1;
124  if (detail::dfu_impl_t::select (jobspec, root, meta, x) < 0) {
125  errno = (errno == EBUSY)? ENODEV : errno;
126  detail::dfu_impl_t::update ();
127  }
128  m_total_preorder += detail::dfu_impl_t::get_preorder_count ();
129  m_total_postorder += detail::dfu_impl_t::get_postorder_count ();
130  }
131  break;
132  }
133  case match_op_t::MATCH_ALLOCATE:
134  errno = EBUSY;
135  break;
136  default:
137  break;
138  }
139 
140 out:
141  errno = (!errno)? saved_errno : errno;
142  return rc;
143 }

240 int dfu_traverser_t::initialize ()
241 {
242  int rc = 0;
243  vtx_t root;
244  if (!get_graph () || !get_graph_db () || !get_match_cb ()) {
245  errno = EINVAL;
246  return -1;
247  }
248 
249  m_initialized = false;
250  detail::dfu_impl_t::reset_color ();
251  for (auto &subsystem : get_match_cb ()->subsystems ()) {
252  std::map<std::string, int64_t> from_dfv;
253  if (get_graph_db ()->metadata.roots.find (subsystem)
254  == get_graph_db ()->metadata.roots.end ()) {
255  errno = ENOTSUP;
256  rc = -1;
257  break;
258  }
259  root = get_graph_db ()->metadata.roots.at(subsystem);
260  rc += detail::dfu_impl_t::prime_pruning_filter (subsystem,
261  root, from_dfv);
262  }
263  m_initialized = (rc == 0)? true : false;
264  return rc;
265 }

277 int dfu_traverser_t::run (Jobspec::Jobspec &jobspec,
278  std::shared_ptr<match_writers_t> &writers,
279  match_op_t op, int64_t jobid, int64_t *at)
280 {
281  const subsystem_t &dom = get_match_cb ()->dom_subsystem ();
282  graph_duration_t graph_duration =
283  get_graph_db ()->metadata.graph_duration;
284  if (!get_graph () || !get_graph_db ()
285  || (get_graph_db ()->metadata.roots.find (dom)
286  == get_graph_db ()->metadata.roots.end ())
287  || !get_match_cb () || jobspec.resources.empty ()) {
288  errno = EINVAL;
289  return -1;
290  }
291 
292  int rc = -1;
293  int64_t graph_end = std::chrono::duration_cast<std::chrono::seconds>
294  (graph_duration.graph_end
295  .time_since_epoch ()).count ();
296  detail::jobmeta_t meta;
297  vtx_t root = get_graph_db ()->metadata.roots.at (dom);
298  bool x = detail::dfu_impl_t::exclusivity (jobspec.resources, root);
299  const std::set<std::string> exclusive_types =
300  detail::dfu_impl_t::get_exclusive_resource_types ();
301  std::unordered_map<std::string, int64_t> dfv;
302 
303  detail::dfu_impl_t::prime_jobspec (jobspec.resources, dfv);
304  if (meta.build (jobspec, detail::jobmeta_t::alloc_type_t::AT_ALLOC, jobid,
305  *at, graph_duration) < 0)
306  return -1;
307 
308  if ( (op == match_op_t::MATCH_SATISFIABILITY)
309  && (rc = is_satisfiable (jobspec, meta, x, root, dfv)) == 0) {
310  detail::dfu_impl_t::update ();
311  } else if ( (rc = schedule (jobspec, meta, x, op, root, dfv)) == 0) {
312  *at = meta.at;
313  if (*at < 0 or *at >= graph_end) {
314  detail::dfu_impl_t::reset_exclusive_resource_types
315  (exclusive_types);
316  errno = EINVAL;
317  return -1;
318  }
319  // If job ends after the resource graph expires, reduce the duration
320  // so it coincides with the graph expiration. Note that we could
321  // arguably return ENOTSUP/EINVAL instead. Also note that we
322  // know *at < graph_end from the previous check, and meta.duration
323  // is < int64_t max from meta.build.
324  if ( (*at + static_cast<int64_t> (meta.duration)) > graph_end)
325  meta.duration = graph_end - *at;
326  // returns 0 or -1
327  rc = detail::dfu_impl_t::update (root, writers, meta);
328  }
329  // returns 0 or -1
330  rc += detail::dfu_impl_t::reset_exclusive_resource_types (exclusive_types);
331 
332  return rc;
333 }
