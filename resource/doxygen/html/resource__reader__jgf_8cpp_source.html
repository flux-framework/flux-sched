269 int resource_reader_jgf_t::unpack_and_remap_vtx (fetch_helper_t &f,
270  json_t *paths,
271  json_t *properties)
272 {
273  json_t *value = NULL;
274  const char *key = NULL;
275  uint64_t remap_rank;
276  uint64_t remap_id;
277 
278  if (namespace_remapper.query_exec_target (
279  static_cast<uint64_t> (f.rank),
280  remap_rank) < 0) {
281  m_err_msg += __FUNCTION__;
282  m_err_msg += ": error remapping rank id=";
283  m_err_msg += std::to_string (f.rank) + ".\n";
284  goto error;
285  }
286  if (remap_rank
287  > static_cast<uint64_t> (std::numeric_limits<int64_t>::max ())) {
288  errno = EOVERFLOW;
289  m_err_msg += __FUNCTION__;
290  m_err_msg += ": remapped rank too large.\n";
291  goto error;
292  }
293  f.set_remapped_rank (static_cast<int64_t> (remap_rank));
294 
295  if (std::string ("core") == f.type) {
296  if (namespace_remapper.query (f.rank, "core", f.id, remap_id) < 0) {
297  m_err_msg += __FUNCTION__;
298  m_err_msg += ": error remapping core id=" + std::to_string (f.id);
299  m_err_msg += " rank=" + std::to_string (f.rank) + ".\n";
300  goto error;
301  }
302  if (remap_id
303  > static_cast<uint64_t> (std::numeric_limits<int>::max ())) {
304  errno = EOVERFLOW;
305  m_err_msg += __FUNCTION__;
306  m_err_msg += ": remapped id too large.\n";
307  goto error;
308  }
309  f.set_remapped_id (static_cast<int> (remap_id));
310  f.set_remapped_name (f.basename + std::to_string (remap_id));
311 
312  json_object_foreach (paths, key, value) {
313  std::string path = json_string_value (value);
314  std::size_t sl = path.find_last_of ("/");
315  if (sl == std::string::npos || path.substr (sl+1, 4) != "core") {
316  errno = EINVAL;
317  m_err_msg += __FUNCTION__;
318  m_err_msg += ": malformed path for core id=";
319  m_err_msg += std::to_string (f.id) + ".\n";
320  goto error;
321  }
322  f.paths[std::string (key)] = path.substr (0, sl+1)
323  + f.basename
324  + std::to_string (remap_id);
325  }
326  } else {
327  json_object_foreach (paths, key, value) {
328  f.paths[std::string (key)] = json_string_value (value);
329  }
330  }
331 
332  json_object_foreach (properties, key, value) {
333  f.properties[std::string (key)]
334  = std::string (json_string_value (value));
335  }
336  return 0;
337 error:
338  return -1;
339 }

363 int resource_reader_jgf_t::fill_fetcher (json_t *element, fetch_helper_t &f,
364  json_t **paths, json_t **properties)
365 {
366  int rc = -1;
367  json_t *p = NULL;
368  json_t *metadata = NULL;
369 
370  if ( (json_unpack (element, "{ s:s }", "id", &f.vertex_id) < 0)) {
371  errno = EINVAL;
372  m_err_msg += __FUNCTION__;
373  m_err_msg += ": JGF vertex id key is not found in a node.\n";
374  goto done;
375  }
376  if ( (metadata = json_object_get (element, "metadata")) == NULL) {
377  errno = EINVAL;
378  m_err_msg += __FUNCTION__;
379  m_err_msg += ": key (metadata) is not found in an JGF node for ";
380  m_err_msg += std::string (f.vertex_id) + ".\n";
381  goto done;
382  }
383  if ( (json_unpack (metadata, "{ s:s s:s s:s s:I s:I s:I s?:i s:b s:s s:I }",
384  "type", &f.type, "basename", &f.basename,
385  "name", &f.name, "id", &f.id,
386  "uniq_id", &f.uniq_id, "rank", &f.rank,
387  "status", &f.status, "exclusive", &f.exclusive,
388  "unit", &f.unit, "size", &f.size)) < 0) {
389  errno = EINVAL;
390  m_err_msg += __FUNCTION__;
391  m_err_msg += ": malformed metadata in an JGF node for ";
392  m_err_msg += std::string (f.vertex_id) + "\n";
393  goto done;
394  }
395  if ( (p = json_object_get (metadata, "paths")) == NULL) {
396  errno = EINVAL;
397  m_err_msg += __FUNCTION__;
398  m_err_msg += ": key (paths) does not exist in an JGF node for ";
399  m_err_msg += std::string (f.vertex_id) + ".\n";
400  goto done;
401  }
402  *properties = json_object_get (metadata, "properties");
403  *paths = p;
404  if (*properties && !json_is_object (*properties)) {
405  errno = EINVAL;
406  m_err_msg += __FUNCTION__;
407  m_err_msg += ": key (properties) must be an object or null for ";
408  m_err_msg += std::string (f.vertex_id) + ".\n";
409  goto done;
410  }
411  rc = 0;
412 done:
413  return rc;
414 }
