106 unsigned int matcher_util_api_t::calc_count (
107  const Flux::Jobspec::Resource &resource,
108  unsigned int qc) const
109 {
110  if (resource.count.min > resource.count.max
111  || resource.count.min > qc)
112  return 0;
113 
114  unsigned int range = 0;
115  unsigned int count = 0;
116  unsigned int cur = resource.count.min;
117 
118  switch (resource.count.oper) {
119  case '+':
120  count = (qc < resource.count.max)? qc : resource.count.max;
121  range = count - cur;
122  count -= (range % resource.count.operand);
123  break;
124  case '*':
125  if (resource.count.operand == 1)
126  count = cur;
127  else {
128  while (cur <= qc && cur <= resource.count.max) {
129  count = cur;
130  cur *= resource.count.operand;
131  }
132  }
133  break;
134  case '^':
135  if (resource.count.operand < 2)
136  count = cur;
137  else {
138  while (cur <= qc && cur <= resource.count.max) {
139  unsigned int base = cur;
140  count = cur;
141  for (int i = 1;
142  i < resource.count.operand; i++)
143  cur *= base;
144  }
145  }
146  break;
147  default:
148  break;
149  }
150  return count;
151 }

159 int matcher_util_api_t::register_resource_pair (const std::string &subsystem,
160  std::string &r_pair)
161 {
162  int rc = -1;
163  size_t pos = 0;
164  std::string h;
165  std::string l;
166  std::string split = ":";
167 
168  if ((pos = r_pair.find (split)) == std::string::npos)
169  goto done;
170  h = r_pair.substr (0, pos);
171  h.erase (std::remove_if (h.begin (), h.end (), ::isspace), h.end ());
172  if (h.empty ()) {
173  errno = EINVAL;
174  goto done;
175  }
176  if (h == "ALL")
177  h = ANY_RESOURCE_TYPE;
178  l = r_pair.erase (0, pos + split.length ());
179  l.erase (std::remove_if (l.begin (), l.end (), ::isspace), l.end ());
180  if (l.empty ()) {
181  errno = EINVAL;
182  goto done;
183  }
184  set_pruning_type (subsystem, h, l);
185  rc = 0;
186 
187 done:
188  return rc;
189 }

191 int matcher_util_api_t::set_pruning_types_w_spec (const std::string &subsystem,
192  const std::string &spec)
193 {
194  int rc = -1;
195  size_t pos = 0;
196  std::string spec_copy = spec;
197  std::string sep = ",";
198 
199  try {
200  while ((pos = spec_copy.find (sep)) != std::string::npos) {
201  std::string r_pair = spec_copy.substr (0, pos);
202  if (register_resource_pair (subsystem, r_pair) < 0)
203  goto done;
204  spec_copy.erase (0, pos + sep.length ());
205  }
206  if (register_resource_pair (subsystem, spec_copy) < 0)
207  goto done;
208  rc = 0;
209  } catch (std::out_of_range &e) {
210  errno = EINVAL;
211  rc = -1;
212  } catch (std::bad_alloc &e) {
213  errno = ENOMEM;
214  rc = -1;
215  }
216 
217 done:
218  return rc;

221 void matcher_util_api_t::set_pruning_type (const std::string &subsystem,
222  const std::string &anchor_type,
223  const std::string &prune_type)
224 {
225  // Use operator[] to create an entry if subsystem key doesn't exist
226  auto &s = m_pruning_types[subsystem];
227  if (anchor_type == ANY_RESOURCE_TYPE) {
228  // Check whether you have already installed prune_type.
229  // If so, remove it as you want to install it against ANY_RESOURCE_TYPE.
230  for (auto &kv : s)
231  kv.second.erase (prune_type);
232  // final container is "set" so it will only allow unique prune_types
233  s[anchor_type].insert (prune_type);
234  } else {
235  if ((s.find (ANY_RESOURCE_TYPE) != s.end ())) {
236  auto &prune_set = s[ANY_RESOURCE_TYPE];
237  if (prune_set.find (prune_type) == prune_set.end ()) {
238  // If prune_type does not exist against ANY_RESOURCE_TYPE
239  // Install it against anchor_type, an individual resource type
240  s[anchor_type].insert (prune_type);
241  } // orelse NOOP
242  } else {
243  s[anchor_type].insert (prune_type);
244  }
245  }
246  m_total_set[subsystem].insert (prune_type);
247 }
