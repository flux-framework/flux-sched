31 int resource_reader_hwloc_t::check_hwloc_version (std::string &m_err_msg)
32 {
33  unsigned int hwloc_version = hwloc_get_api_version ();
34 
35  if ((hwloc_version >> 16) != (HWLOC_API_VERSION >> 16)) {
36  std::stringstream msg;
37  msg << "Compiled for hwloc API 0x"
38  << std::hex << HWLOC_API_VERSION
39  << " but running on library API 0x"
40  << hwloc_version << "; ";
41  m_err_msg += msg.str ();

43  return -1;
44  }
45  return 0;

64 vtx_t resource_reader_hwloc_t::add_new_vertex (resource_graph_t &g,
65  resource_graph_metadata_t &m,
66  const vtx_t &parent, int64_t id,
67  const std::string &subsys,
68  const std::string &type,
69  const std::string &basename,
70  const std::string &name,
71  const std::map<std::string, std::string> &properties,
72  int size, int rank)
73 {
74  vtx_t v = boost::add_vertex (g);
75 
76  // Set properties of the new vertex
77  bool is_root = false;
78  if (parent == boost::graph_traits<resource_graph_t>::null_vertex ())
79  is_root = true;
80  std::string istr = (id != -1)? std::to_string (id) : "";
81  std::string prefix = is_root ? "" : g[parent].paths[subsys];
82 
83  g[v].type = type;
84  g[v].basename = basename;
85  g[v].size = size;
86  g[v].uniq_id = v;
87  g[v].rank = rank;
88  g[v].schedule.plans = planner_new (0, INT64_MAX, size, type.c_str ());
89  g[v].idata.x_checker = planner_new (0, INT64_MAX,
90  X_CHECKER_NJOBS, X_CHECKER_JOBS_STR);
91  g[v].id = id;
92  g[v].name = (name != "")? name : basename + istr;
93  g[v].paths[subsys] = prefix + "/" + g[v].name;
94  g[v].idata.member_of[subsys] = "*";
95  g[v].status = resource_pool_t::status_t::UP;
96  g[v].properties = properties;
97 
98  // Indexing for fast look-up
99  m.by_path[g[v].paths[subsys]].push_back (v);
100  m.by_type[g[v].type].push_back (v);
101  m.by_name[g[v].name].push_back (v);
102  m.by_rank[rank].push_back (v);
103  return v;
104 }

106 int resource_reader_hwloc_t::add_metadata (resource_graph_metadata_t &m,
107  edg_t e, vtx_t src, vtx_t tgt,
108  resource_graph_t &g)
109 {
110  // add this edge to by_outedges metadata
111  auto iter = m.by_outedges.find (src);
112  if (iter == m.by_outedges.end ()) {
113  auto ret = m.by_outedges.insert (
114  std::make_pair (
115  src,
116  std::map<std::pair<uint64_t, int64_t>, edg_t,
117  std::greater<
118  std::pair<uint64_t,
119  int64_t>>> ()));
120  if (!ret.second) {
121  errno = ENOMEM;
122  m_err_msg += "error creating out-edge metadata map: "
123  + g[src].name + " -> " + g[tgt].name + "; ";
124  return -1;
125  }
126  iter = m.by_outedges.find (src);
127  }
128  std::pair<uint64_t, int64_t> key = std::make_pair (
129  g[e].idata.get_weight (),
130  g[tgt].uniq_id);
131  auto ret = iter->second.insert (std::make_pair (key, e));
132  if (!ret.second) {
133  errno = ENOMEM;
134  m_err_msg += "error inserting an edge to out-edge metadata map: "
135  + g[src].name + " -> " + g[tgt].name + "; ";
136  return -1;
137  }
138  return 0;
139 }

141 int resource_reader_hwloc_t::walk_hwloc (resource_graph_t &g,
142  resource_graph_metadata_t &m,
143  const hwloc_topology_t topo,
144  const hwloc_obj_t obj,
145  const vtx_t parent, int rank)
146 {
147  bool supported_resource = true;
148  std::string type, basename;
149  std::string name = "";
150  int64_t id = obj->logical_index;
151  int rc = 0;
152  unsigned int size = 1;
153  std::map<std::string, std::string> properties;
154 
155  switch(obj->type) {
156  case HWLOC_OBJ_MACHINE: {
157  // TODO: add signature to support multiple ranks per node
158  const char *hwloc_name = hwloc_obj_get_info_by_name (obj, "HostName");
159  if (!hwloc_name || !in_allowlist ("node")) {
160  supported_resource = false;
161  break;
162  }
163  type = "node";
164  name = hwloc_name;
165  if (split_hostname (name, basename, id) < 0) {
166  m_err_msg += "Error converting node id for " + name + "; ";
167  rc = -1;
168  }
169  break;
170  }
171  case HWLOC_OBJ_GROUP: {
172  if (!in_allowlist ("group")) {
173  supported_resource = false;
174  break;
175  }
176  type = "group";
177  basename = type;
178  break;
179  }
180  case HWLOC_OBJ_NUMANODE: {
181  if (!in_allowlist ("numanode")) {
182  supported_resource = false;
183  break;
184  }
185  type = "numanode";
186  basename = type;
187  break;
188  }
189  case HWLOC_OBJ_PACKAGE: {
190  if (!in_allowlist ("socket")) {
191  supported_resource = false;
192  break;
193  }
194  type = "socket";
195  basename = type;
196  break;
197  }
198 #if HWLOC_API_VERSION < 0x00020000
199  case HWLOC_OBJ_CACHE: {
200  std::string r = "L" + std::to_string (obj->attr->cache.depth) + "cache";
201  if (!in_allowlist (r)) {
202  supported_resource = false;
203  break;
204  }
205  type = "cache";
206  basename = r;
207  size = obj->attr->cache.size / 1024;
208  break;
209  }
210 #else
211  case HWLOC_OBJ_L1CACHE: {
212  if (!in_allowlist ("L1cache")) {
213  supported_resource = false;
214  break;
215  }
216  type = "cache";
217  basename = "L1" + type;
218  size = obj->attr->cache.size / 1024;
219  break;
220  }
221  case HWLOC_OBJ_L2CACHE: {
222  if (!in_allowlist ("L2cache")) {
223  supported_resource = false;
224  break;
225  }
226  type = "cache";
227  basename = "L2" + type;
228  size = obj->attr->cache.size / 1024;
229  break;
230  }
231  case HWLOC_OBJ_L3CACHE: {
232  if (!in_allowlist ("L3cache")) {
233  supported_resource = false;
234  break;
235  }
236  type = "cache";
237  basename = "L3" + type;
238  size = obj->attr->cache.size / 1024;
239  break;
240  }
241 #endif
242  case HWLOC_OBJ_CORE: {
243  if (!in_allowlist ("core")) {
244  supported_resource = false;
245  break;
246  }
247  type = "core";
248  basename = type;
249  break;
250  }
251  case HWLOC_OBJ_PU: {
252  if (!in_allowlist ("pu")) {
253  supported_resource = false;
254  break;
255  }
256  type = "pu";
257  basename = type;
258  break;
259  }
260  case HWLOC_OBJ_OS_DEVICE: {
261  supported_resource = false;
262  if (!obj->attr)
263  break;
264 
265  /* Detect any COPROC as GPU device.
266  * Also, some AMD GPUs are detected as type == GPU with
267  * subtype RSMI (rsmiX), detect those also:
268  */
269  if (obj->attr->osdev.type == HWLOC_OBJ_OSDEV_COPROC
270  || (obj->attr->osdev.type == HWLOC_OBJ_OSDEV_GPU
271  && strncmp (obj->name, "rsmi", 4) == 0)) {
272  if (!in_allowlist ("gpu")) {
273  break;
274  }
275  /* hwloc doesn't provide the logical index only amongst CoProc
276  * devices so we parse this info from the name until hwloc provide
277  * better support.
278  */
279  if (strncmp (obj->name, "cuda", 4) == 0)
280  id = atoi (obj->name + 4);
281  else if (strncmp (obj->name, "rsmi", 4) == 0)
282  id = atoi (obj->name + 4);
283  else if (strncmp (obj->name, "opencl", 6) == 0) {
284  /* Naming convention of opencl devices for hwloc:
285  * "opencl" followed by a platform ID followed by a device ID.
286  * Then, the letter d delimits platform id and device id.
287  */
288  const char *delim = strchr (obj->name + 6, 'd');
289  id = atoi (delim + 1);
290  }
291  if (namespace_remapper.is_remapped ()) {
292  uint64_t remap_id;
293  if (namespace_remapper.query (rank, "gpu", id, remap_id) < 0) {
294  m_err_msg += "Error remapping gpu id="
295  + std::to_string (id) +"; ";
296  rc = -1;
297  break;
298  }
299  if (remap_id
300  > static_cast<uint64_t> (std::numeric_limits<int>::max ())) {
301  errno = EOVERFLOW;
302  m_err_msg += "Remapped gpu id too large; ";
303  rc = -1;
304  break;
305  }
306  id = static_cast<int> (remap_id);
307  }
308  type = "gpu";
309  basename = type;
310  supported_resource = true;
311  } else if (obj->attr->osdev.type == HWLOC_OBJ_OSDEV_BLOCK) {
312  if (!in_allowlist ("storage")) {
313  break;
314  }
315  // Size from hwloc is in kBs (base 10)
316  const char* size_str = hwloc_obj_get_info_by_name (obj, "Size");
317  if (size_str == NULL) {
318  errno = ENOENT;
319  m_err_msg += "Error getting the size of block storage device; ";
320  rc = -1;
321  break;
322  }
323  long int num_bytes = strtol (size_str, NULL, 10);
324  if (num_bytes == 0 || num_bytes == LONG_MIN || num_bytes == LONG_MAX) {
325  // strtol sets errno to ERANGE
326  m_err_msg += "Error parsing the size of block storage device; ";
327  rc = -1;
328  break;
329  }
330  size = num_bytes / 1000000; // kBs -> gBs
331  type = "storage";
332  basename = type;
333  supported_resource = true;
334 
335  const char* devID_str = hwloc_obj_get_info_by_name (obj, "LinuxDeviceID");
336  if (devID_str == NULL) {
337  devID_str = "-1";
338  }
339  auto ret = properties.insert (std::pair<std::string, std::string>("LinuxDeviceID", devID_str));
340  if (!ret.second) {
341  errno = EEXIST;
342  m_err_msg += "Error inserting LinuxDeviceID into properties map; ";
343  rc = -1;
344  break;
345  }
346  }
347  break;
348  }
349  default: {
350  supported_resource = false;
351  break;
352  }
353  }
354 
355  // A valid ancestor vertex to pass to the recursive call
356  vtx_t valid_ancestor;
357  if (!supported_resource) {
358  valid_ancestor = parent;
359  } else {
360  const std::string subsys = "containment";
361  vtx_t v = add_new_vertex (g, m, parent,
362  id, subsys, type, basename,
363  name, properties, size, rank);
364  valid_ancestor = v;
365  std::string relation = "contains";
366  std::string rev_relation = "in";
367  edg_t e;
368  bool inserted; // set to false when we try and insert a parallel edge
369 
370  tie (e, inserted) = add_edge (parent, v, g);
371  if (!inserted) {
372  errno = ENOMEM;
373  m_err_msg += "error inserting a new edge: "
374  + g[parent].name + " -> " + g[v].name + "; ";
375  return -1;
376  }
377  g[e].idata.member_of[subsys] = relation;
378  g[e].name[subsys] = relation;
379  if (add_metadata (m, e, parent, v, g) < 0)
380  return -1;
381 
382  tie (e, inserted) = add_edge (v, parent, g);
383  if (!inserted) {
384  errno = ENOMEM;
385  m_err_msg += "error inserting a new edge: "
386  + g[v].name + " -> " + g[parent].name + "; ";
387  return -1;
388  }
389  g[e].idata.member_of[subsys] = rev_relation;
390  g[e].name[subsys] = rev_relation;
391  if (add_metadata (m, e, v, parent, g) < 0)
392  return -1;
393  }
394 
395  hwloc_obj_t curr_child = NULL;
396  while ((curr_child = hwloc_get_next_child (topo, obj, curr_child)) != NULL) {
397  rc += walk_hwloc (g, m, topo, curr_child, valid_ancestor, rank);
398  }
399 
400  return rc;
401 }

403 int resource_reader_hwloc_t::unpack_internal (resource_graph_t &g,
404  resource_graph_metadata_t &m,
405  vtx_t &vtx,
406  const std::string &str, int rank)
407 {
408  int rc = -1;
409  size_t len = str.length ();
410  hwloc_topology_t topo;
411  hwloc_obj_t hwloc_root;
412 
413  if ( hwloc_topology_init (&topo) != 0 ) {
414  errno = ENOMEM;
415  m_err_msg += "Error initializing hwloc topology; ";
416  goto done;
417  }
418 #if HWLOC_API_VERSION < 0x20000
419  if ( hwloc_topology_set_flags (topo, HWLOC_TOPOLOGY_FLAG_IO_DEVICES) != 0) {
420  errno = EINVAL;
421  m_err_msg += "Error setting hwloc topology flag; ";
422  goto done;
423  }
424 #else
425  if (hwloc_topology_set_io_types_filter (topo,
426  HWLOC_TYPE_FILTER_KEEP_IMPORTANT)
427  < 0) {
428  errno = EINVAL;
429  m_err_msg += "hwloc_topology_set_io_types_filter; ";
430  goto done;
431  }
432  if (hwloc_topology_set_cache_types_filter (topo,
433  HWLOC_TYPE_FILTER_KEEP_STRUCTURE)
434  < 0) {
435  errno = EINVAL;
436  m_err_msg += "hwloc_topology_set_cache_types_filter; ";
437  goto done;
438  }
439  if (hwloc_topology_set_icache_types_filter (topo,
440  HWLOC_TYPE_FILTER_KEEP_STRUCTURE)
441  < 0) {
442  errno = EINVAL;
443  m_err_msg += "hwloc_topology_set_icache_types_filter; ";
444  goto done;
445  }
446 #endif
447  if ( hwloc_topology_set_xmlbuffer (topo, str.c_str (), len) != 0) {
448  errno = EINVAL;
449  m_err_msg += "Error setting xmlbuffer; ";
450  goto done;
451  }
452  if ( hwloc_topology_load (topo) != 0) {
453  hwloc_topology_destroy (topo);
454  m_err_msg += "Error hwloc load: rank " + std::to_string (rank) + "; ";
455  goto done;
456  }
457 
458  hwloc_root = hwloc_get_root_obj (topo);
459  if ((rc = walk_hwloc (g, m, topo, hwloc_root, vtx, rank)) < 0) {
460  hwloc_topology_destroy (topo);
461  m_err_msg += "Error hwloc walk: rank " + std::to_string (rank) + "; ";
462  goto done;
463  }
464  hwloc_topology_destroy (topo);
465  rc = 0;
466 
467 done:
468  return rc;
469 }
