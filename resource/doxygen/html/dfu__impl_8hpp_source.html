66  int build (Jobspec::Jobspec &jobspec, alloc_type_t alloc, int64_t id,
67  int64_t t, graph_duration_t &graph_duration)
68  {
69  at = t;
70  now = t;
71  jobid = id;
72  alloc_type = alloc;
73  int64_t g_duration = std::chrono::duration_cast<std::chrono::seconds>
74  (graph_duration.graph_end - graph_duration.graph_start).count ();
75 
76  if (g_duration <= 0) {
77  errno = EINVAL;
78  return -1;
79  }
80  // Ensure that duration is shorter than expressible
81  // int64_t max () for comparison with at in dfu_traverser_t::run
82  if ( (jobspec.attributes.system.duration >
83  static_cast<double> (g_duration))
84  || (jobspec.attributes.system.duration >
85  static_cast<double> (std::numeric_limits<int64_t>::max ()))) {
86  errno = EINVAL;
87  return -1;
88  }
89  if (jobspec.attributes.system.duration == 0.0f)
90  duration = g_duration;
91  else
92  duration = (int64_t)jobspec.attributes.system.duration;
93  if (jobspec.attributes.system.queue != "") {
94  m_queue = jobspec.attributes.system.queue;
95  m_queue_set = true;
96  }
97  constraint = jobspec.attributes.system.constraint;
98  return 0;
99  }
100 
101 private:
102  bool m_queue_set = false;
103  std::string m_queue = "";
104 };

506 template <class lookup_t>
507 int dfu_impl_t::count_relevant_types (planner_multi_t *plan,
508  const lookup_t &lookup,
509  std::vector<uint64_t> &resource_counts)
510 {
511  int rc = 0;
512  size_t len = planner_multi_resources_len (plan);
513  const char *type = nullptr;
514  for (unsigned int i = 0; i < len; ++i) {
515  type = planner_multi_resource_type_at (plan, i);
516  if (lookup.find (type) != lookup.end ()) {
517  uint64_t n = (uint64_t)lookup.at (type);
518  resource_counts.push_back (n);
519  } else {
520  resource_counts.push_back (0);
521  }
522  }
523  return rc;
524 }
