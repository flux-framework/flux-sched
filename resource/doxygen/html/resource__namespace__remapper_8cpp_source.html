81 int distinct_range_t::get_low_high (const std::string &exec_target_range,
82  uint64_t &low, uint64_t &high)
83 {
84  try {
85  long int n;
86  size_t ndash;
87  std::string exec_target;
88  std::istringstream istr{exec_target_range};
89  std::vector<uint64_t> targets;
90 
91  if ( (ndash = std::count (exec_target_range.begin (),
92  exec_target_range.end (), '-')) > 1)
93  goto inval;
94  while (std::getline (istr, exec_target, '-')) {
95  if ( (n = std::stol (exec_target)) < 0)
96  goto inval;
97  targets.push_back (static_cast<uint64_t> (n));
98  }
99  low = high = targets[0];
100  if (targets.size () == 2)
101  high = targets[1];
102  if (low > high)
103  goto inval;
104  } catch (std::invalid_argument &) {
105  goto inval;
106  } catch (std::out_of_range &) {
107  errno = ERANGE;
108  goto error;
109  }
110  return 0;
111 inval:
112  errno = EINVAL;
113 error:
114  return -1;
115 }

124 int resource_namespace_remapper_t::add (const uint64_t low, const uint64_t high,
125  const std::string &name_type,
126  uint64_t ref_id, uint64_t remapped_id)
127 {
128  try {
129  const distinct_range_t exec_target_range{low, high};
130  auto m_remap_iter = m_remap.find (exec_target_range);
131 
132  if (m_remap_iter == m_remap.end ()) {
133  m_remap.emplace (exec_target_range,
134  std::map<const std::string,
135  std::map<uint64_t, uint64_t>> ());
136  } else if (exec_target_range != m_remap_iter->first)
137  goto inval; // key must be exact
138 
139  if (m_remap[exec_target_range].find (name_type)
140  == m_remap[exec_target_range].end ())
141  m_remap[exec_target_range][name_type] = std::map<uint64_t,
142  uint64_t> ();
143  if (m_remap[exec_target_range][name_type].find (ref_id)
144  != m_remap[exec_target_range][name_type].end ()) {
145  errno = EEXIST;
146  goto error;
147  }
148  m_remap[exec_target_range][name_type][ref_id] = remapped_id;
149  } catch (std::bad_alloc &) {
150  errno = ENOMEM;
151  goto error;
152  } catch (std::invalid_argument &) {
153  goto inval;
154  }
155  return 0;
156 inval:
157  errno = EINVAL;
158 error:
159  return -1;
160 }

180 int resource_namespace_remapper_t::add_exec_target_range (
181  const std::string &exec_target_range,
182  const distinct_range_t &remapped_exec_target_range)
183 {
184  try {
185  uint64_t low, high, r_low, r_high, i, j;
186  if (distinct_range_t::get_low_high (exec_target_range, low, high) < 0)
187  goto error;
188  r_low = remapped_exec_target_range.get_low ();
189  r_high = remapped_exec_target_range.get_high ();
190  if ((high - low) != (r_high - r_low))
191  goto inval;
192  for (i = low, j = r_low; i <= high && j <= r_high; i++, j++) {
193  if (add (exec_target_range, "exec-target", i, j) < 0)
194  goto error;
195  }
196  } catch (std::bad_alloc &) {
197  errno = ENOMEM;
198  goto error;
199  }
200  return 0;
201 inval:
202  errno = EINVAL;
203 error:
204  return -1;
205 }
