136 scheduled_point_t *scheduled_point_tree_t::get_state (int64_t at) const
137 {
138  scheduled_point_t *last_state = nullptr;
139  scheduled_point_rb_node_t *node = m_tree.get_root ();
140  while (node) {
141  scheduled_point_t *this_data = nullptr;
142  this_data = node->get_point ();
143  int64_t result = at - this_data->at;
144  if (result < 0) {
145  node = node->get_left ();
146  } else if (result > 0) {
147  last_state = get_recent_state (this_data, last_state);
148  node = node->get_right ();
149  } else {
150  return this_data;
151  }
152  }
153  return last_state;
154 }
