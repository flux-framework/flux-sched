93 extern "C" int reapi_cli_match (reapi_cli_ctx_t *ctx,
94  match_op_t match_op, const char *jobspec,
95  uint64_t *jobid, bool *reserved,
96  char **R, int64_t *at, double *ov)
97 {
98  int rc = -1;
99  std::string R_buf = "";
100  char *R_buf_c = nullptr;
101 
102  if (!ctx || !ctx->rqt) {
103  errno = EINVAL;
104  goto out;
105  }
106 
107  *jobid = ctx->rqt->get_job_counter ();
108  if ((rc = reapi_cli_t::match_allocate (ctx->rqt, match_op,
109  jobspec, *jobid, *reserved,
110  R_buf, *at, *ov)) < 0) {
111  goto out;
112  }
113 
114  if ( !(R_buf_c = strdup (R_buf.c_str ()))) {
115  ctx->err_msg = __FUNCTION__;
116  ctx->err_msg += ": ERROR: can't allocate memory\n";
117  errno = ENOMEM;
118  rc = -1;
119  goto out;
120  }
121  (*R) = R_buf_c;
122 
123 out:
124  return rc;
125 }

140  const char *jobspec,

142 {

144  uint64_t jobid;
145  bool reserved;
146  char *R;
147  int64_t at;

151  ret = reapi_cli_match (ctx, match_op, jobspec, &jobid,
152  &reserved, &R, &at, ov);

154  // check for satisfiability

156  *sat = false;
157 
158  return ret;
