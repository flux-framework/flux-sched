266 int resource_reader_rv1exec_t::update_vertex (resource_graph_t &g,
267  vtx_t vtx,
268  updater_data &update_data)
269 {
270 
271  int rc = -1;
272  int64_t span = -1;
273  int64_t avail = -1;
274  planner_t *plans = NULL;
275 
276  // Check and update plan
277  if ( (plans = g[vtx].schedule.plans) == NULL) {
278  errno = EINVAL;
279  m_err_msg += __FUNCTION__;
280  m_err_msg += ": plan for " + g[vtx].name + " is null.\n";
281  goto error;
282  }
283  if ( (avail = planner_avail_resources_during (plans,
284  update_data.at,
285  update_data.duration)) == -1) {
286  m_err_msg += __FUNCTION__;
287  m_err_msg += ": planner_avail_resource_during return -1 for ";
288  m_err_msg += g[vtx].name + ".\n";
289  goto error;
290  }
291  if (avail < g[vtx].size) {
292  // if g[v] has already been allocated/reserved, this is an error
293  m_err_msg += __FUNCTION__;
294  m_err_msg += ": " + g[vtx].name + " is unavailable.\n";
295  goto error;
296  }
297  // Update the vertex plan here (not in traverser code).
298  // Traverser update () will handle aggregate filters and
299  // exclusivity checking filter.
300  // Can't update the rank-level (node) vertex yet- we
301  // don't know if all its children are allocated.
302  // Note this is a hard-coded option. Support for more flexible
303  // types may require extending rv1exec.
304  if (g[vtx].type == "node")
305  return 0;
306  // Name is anything besides node
307  if ( (span = planner_add_span (plans,
308  update_data.at,
309  update_data.duration,
310  static_cast<const uint64_t> (g[vtx].size))) == -1) {
311  m_err_msg += __FUNCTION__;
312  m_err_msg += ": can't add span into " + g[vtx].name + ".\n";
313  goto error;
314  }
315 
316  if (update_data.reserved)
317  g[vtx].schedule.reservations[update_data.jobid] = span;
318  else
319  g[vtx].schedule.allocations[update_data.jobid] = span;
320 
321  update_data.updated_vertices[g[vtx].rank].push_back (vtx);
322 
323  rc = 0;
324 
325 error:
326  return rc;
327 }

329 int resource_reader_rv1exec_t::undo_vertices (resource_graph_t &g,
330  updater_data &update_data)
331 {
332 
333  int rc = -1;
334  int64_t span = -1;
335  planner_t *plans = NULL;
336 
337  for (auto &[rank, vertices] : update_data.updated_vertices) {
338  for (vtx_t vtx : vertices) {
339  // Check plan
340  if ( (plans = g[vtx].schedule.plans) == NULL) {
341  errno = EINVAL;
342  m_err_msg += __FUNCTION__;
343  m_err_msg += ": plan for " + g[vtx].name + " is null.\n";
344  goto error;
345  }
346  // Remove job tags
347  if (update_data.reserved) {
348  span = g[vtx].schedule.reservations.at (update_data.jobid);
349  g[vtx].schedule.reservations.erase (update_data.jobid);
350  } else {
351  span = g[vtx].schedule.allocations.at (update_data.jobid);
352  g[vtx].schedule.allocations.erase (update_data.jobid);
353  }
354  // Remove the span.
355  if (planner_rem_span (plans, span) == -1) {
356  m_err_msg += __FUNCTION__;
357  m_err_msg += ": can't remove span from " + g[vtx].name + ".\n";
358  goto error;
359  }
360  }
361  }
362 
363  rc = 0;
364 
365 error:
366  return rc;
367 }

369 int resource_reader_rv1exec_t::update_edges (resource_graph_t &g,
370  resource_graph_metadata_t &m,
371  vtx_t src, vtx_t dst,
372  updater_data &update_data)
373 {
374  edg_t e;

377  boost::graph_traits<resource_graph_t>::out_edge_iterator ei, ei_end;
378 
379  boost::tie (ei, ei_end) = boost::out_edges (src, g);
380  for (; ei != ei_end; ++ei) {
381  if (boost::target (*ei, g) == dst) {
382  e = *ei;

384  break;
385  }
386  }
387  if (!found) {
388  errno = EINVAL;
389  m_err_msg += __FUNCTION__;
390  m_err_msg += ": rv1exec edge not found in resource graph.\n";
391  goto error;
392  }
393  g[e].idata.set_for_trav_update (g[dst].size,
394  true, update_data.token);
395 
396  return 0;
397 
398 error:
399  return -1;

402 int resource_reader_rv1exec_t::update_exclusivity (resource_graph_t &g,
403  resource_graph_metadata_t &m,
404  vtx_t vtx,
405  updater_data &update_data)
406 {
407  // idata tag and exclusive checker update
408  int64_t span = -1;
409  planner_t *plans = NULL;
410 
411  const auto &rank_ex = update_data.updated_vertices.find (g[vtx].rank);
412  if (rank_ex == update_data.updated_vertices.end ()) {
413  m_err_msg += __FUNCTION__;
414  m_err_msg += ": rank not found in agfilters map.\n";
415  return -1;
416  }
417  // This check enforces a rigid constraint on rank equivalence
418  // between graph initialization and rank in rv1exec string.
419  const auto &by_rank = m.by_rank.find (g[vtx].rank);
420  if (by_rank == m.by_rank.end ()) {
421  m_err_msg += __FUNCTION__;
422  m_err_msg += ": rank not found in by_rank graph map.\n";
423  return -1;
424  }
425 
426  // If all child vertices allocated, allocate this vertex.
427  // Subtract one since the current node hasn't been added to the
428  // updated_vertices map.
429  if (rank_ex->second.size () != (by_rank->second.size () - 1))
430  return 0;
431  // Counts indicate exclusive
432  if ( (plans = g[vtx].schedule.plans) == NULL) {
433  errno = EINVAL;
434  m_err_msg += __FUNCTION__;
435  m_err_msg += ": plan for " + g[vtx].name + " is null.\n";
436  return -1;
437  }
438  // Update the vertex plan here (not in traverser code).
439  // Traverser update () will handle aggregate filters and
440  // exclusivity checking filter.
441  if ( (span = planner_add_span (plans, update_data.at, update_data.duration,
442  static_cast<const uint64_t> (g[vtx].size))) == -1) {
443  m_err_msg += __FUNCTION__;
444  m_err_msg += ": can't add span into " + g[vtx].name + ".\n";
445  return -1;
446  }
447  // Add job tags
448  if (update_data.reserved)
449  g[vtx].schedule.reservations[update_data.jobid] = span;
450  else
451  g[vtx].schedule.allocations[update_data.jobid] = span;
452  // Add to the updated vertices vector to undo upon error.
453  update_data.updated_vertices[g[vtx].rank].push_back (vtx);
454 
455  return 0;
456 }

458 vtx_t resource_reader_rv1exec_t::add_or_update (resource_graph_t &g,
459  resource_graph_metadata_t &m,
460  vtx_t parent, int64_t id,
461  const std::string &subsys,
462  const std::string &type,
463  const std::string &basename,
464  const std::string &name,

501  vtx = find_vertex (g, m, parent, id, subsys, type, basename,

508  return null_vtx;

511  return null_vtx;
512  // Must be the containment subsystem
513  if (update_edges (g, m, parent, vtx, update_data) == -1)
514  return null_vtx;
515  }
516 
517  return vtx;

520 int resource_reader_rv1exec_t::build_rmap (json_t *rlite,
521  std::map<unsigned, unsigned> &rmap)
522 {
523  int i;
524  size_t index;
525  unsigned rank;
526  json_t *entry = nullptr;
527  const char *ranks = nullptr;
528  struct idset *ids = nullptr;
529  struct idset *r_ids = nullptr;
530 
531  if (!rlite) {
532  errno = EINVAL;
533  goto error;
534  }
535  if ( !(ids = idset_create (0, IDSET_FLAG_AUTOGROW)))
536  goto error;
537 
538  // Create a global rank set that combines all ranks from R_lite array
539  json_array_foreach (rlite, index, entry) {
540  if (json_unpack (entry, "{s:s}", "rank", &ranks) < 0) {
541  errno = EINVAL;
542  goto error;
543  }
544  if ( !(r_ids = idset_decode (ranks)))
545  goto error;
546 
547  rank = idset_first (r_ids);
548  while (rank != IDSET_INVALID_ID) {
549  if (idset_set (ids, rank) < 0)
550  goto error;
551 
552  rank = idset_next (r_ids, rank);
553  }
554  idset_destroy (r_ids);
555  }
556 
557  // A rank map mapping mononically increasing ranks to consecutive IDs from 0
558  i = 0;
559  rank = idset_first (ids);
560  while (rank != IDSET_INVALID_ID) {
561  if (rmap.find (rank) != rmap.end ()) {
562  errno = EEXIST;
563  m_err_msg += __FUNCTION__;
564  m_err_msg += "rmap has no rank=" + std::to_string (rank) + "; ";
565  goto error;
566  }
567 
568  auto res = rmap.insert (std::pair<unsigned, unsigned> (rank, i));
569  if (!res.second) {
570  errno = ENOMEM;
571  goto error;
572  }
573 
574  rank = idset_next (ids, rank);
575  i++;
576  }
577  idset_destroy (ids);
578  return 0;
579 
580 error:
581  idset_destroy (r_ids);
582  idset_destroy (ids);
583  return -1;
584 }
