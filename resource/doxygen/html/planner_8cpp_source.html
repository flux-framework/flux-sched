136 bool planner::operator== (const planner &o) const
137 {
138  if (m_total_resources != o.m_total_resources)
139  return false;
140  if (m_resource_type != o.m_resource_type)
141  return false;
142  if (m_plan_start != o.m_plan_start)
143  return false;
144  if (m_plan_end != o.m_plan_end)
145  return false;
146  if (m_avail_time_iter_set != o.m_avail_time_iter_set)
147  return false;
148  if (m_span_counter != o.m_span_counter)
149  return false;
150  // m_p0 or o.m_p0 could be uninitialized
151  if (m_p0 && o.m_p0) {
152  if (*m_p0 != *(o.m_p0))
153  return false;
154  } else if (m_p0 || o.m_p0) {
155  return false;
156  } // else both nullptr
157  if (!span_lookups_equal (o))
158  return false;
159  if (!avail_time_iters_equal (o))
160  return false;
161  if (!trees_equal (o))
162  return false;
163  return true;
164 }

232 int planner::update_total (uint64_t resource_total)
233 {
234  int64_t delta = resource_total - m_total_resources;
235  int64_t tmp = 0;
236  scheduled_point_t *point = nullptr;
237  if (delta == 0)
238  return 0;
239  m_total_resources = static_cast<int64_t> (resource_total);
240  point = m_sched_point_tree.get_state (m_plan_start);
241  while (point) {
242  // Prevent remaining from taking negative values. This should
243  // reduce likelihood of errors when adding and removing spans.
244  // If the performance penalty is non-negligible we can
245  // investigate letting remaining take negative values.
246  tmp = point->remaining + delta;
247  if (tmp >= 0)
248  point->remaining = tmp;
249  else
250  point->remaining = 0;
251  point = m_sched_point_tree.next (point);
252  }
253  return 0;
254 }

424 int planner::copy_trees (const planner &o)
425 {
426  int rc = 0;
427 
428  // Incoming planner could be empty.
429  if (!o.m_sched_point_tree.empty ()) {
430  // Need to get the state at plan_start, not just m_p0 since they could
431  // have diverged after scheduling.
432  scheduled_point_t *point = o.m_sched_point_tree.get_state (o.m_plan_start);
433  scheduled_point_t *new_point = nullptr;
434  while (point) {
435  new_point = new scheduled_point_t ();
436  new_point->at = point->at;
437  new_point->in_mt_resource_tree = point->in_mt_resource_tree;
438  new_point->new_point = point->new_point;
439  new_point->ref_count = point->ref_count;
440  new_point->scheduled = point->scheduled;
441  new_point->remaining = point->remaining;
442  if ( (rc = m_sched_point_tree.insert (new_point)) != 0)
443  return rc;
444  if ( (rc = m_mt_resource_tree.insert (new_point)) != 0)
445  return rc;
446  point = o.m_sched_point_tree.next (point);
447  }
448  } else {
449  // Erase if incoming planner is empty.
450  rc = erase ();
451  }
452  return rc;
453 }

455 int planner::copy_maps (const planner &o)
456 {
457  int rc = 0;
458 
459  if (!o.m_span_lookup.empty ()) {
460  for (auto const &span_it : o.m_span_lookup) {
461  std::shared_ptr<span_t> new_span = std::make_shared<span_t> ();
462  // Need to deep copy spans, else copies of planners will share
463  // pointers.
464  new_span->start = span_it.second->start;
465  new_span->last = span_it.second->last;
466  new_span->span_id = span_it.second->span_id;
467  new_span->planned = span_it.second->planned;
468  new_span->in_system = span_it.second->in_system;
469  new_span->start_p = m_sched_point_tree.get_state (new_span->start);
470  new_span->last_p = m_sched_point_tree.get_state (new_span->last);
471  m_span_lookup[span_it.first] = new_span;
472  }
473  } else {
474  m_span_lookup.clear ();
475  }
476  if (!o.m_avail_time_iter.empty ()) {
477  for (auto const &avail_it : o.m_avail_time_iter) {
478  // Scheduled point already copied into trees, so fetch
479  // from SP tree.
480  scheduled_point_t *new_avail =
481  m_sched_point_tree.get_state (avail_it.second->at);
482  m_avail_time_iter[avail_it.first] = new_avail;
483  }
484  } else {
485  m_avail_time_iter.clear ();
486  }
487 
488  return rc;
489 }

491 bool planner::span_lookups_equal (const planner &o) const
492 {
493  if (m_span_lookup.size () != o.m_span_lookup.size ())
494  return false;
495  if (!m_span_lookup.empty ()) {
496  // Iterate through indices to use the auto range-based for loop
497  // semantics; otherwise need to create a "zip" function for two maps.
498  for (auto const &this_it : m_span_lookup) {
499  auto const other = o.m_span_lookup.find (this_it.first);
500  if (other == o.m_span_lookup.end ())
501  return false;
502  if (this_it.first != other->first)
503  return false;
504  // Compare span_t
505  if (*(this_it.second) != *(other->second))
506  return false;
507  }
508  }
509  return true;
510 }

512 bool planner::avail_time_iters_equal (const planner &o) const
513 {
514  if (m_avail_time_iter.size () != o.m_avail_time_iter.size ())
515  return false;
516  if (!m_avail_time_iter.empty ()) {
517  // Iterate through indices to use the auto range-based for loop
518  // semantics; otherwise need to create a "zip" function for two maps.
519  for (auto const &this_it : m_avail_time_iter) {
520  auto const other = o.m_avail_time_iter.find (this_it.first);
521  if (other == o.m_avail_time_iter.end ())
522  return false;
523  if (this_it.first != other->first)
524  return false;
525  if (this_it.second && other->second) {
526  if (*(this_it.second) != *(other->second))
527  return false;
528  } else if (this_it.second || other->second) {
529  return false;
530  }
531  }
532  }
533  return true;
534 }

536 bool planner::trees_equal (const planner &o) const
537 {
538  if (m_sched_point_tree.get_size () != o.m_sched_point_tree.get_size ())
539  return false;
540  if (!m_sched_point_tree.empty ()) {
541  scheduled_point_t *this_pt =
542  m_sched_point_tree.get_state (m_plan_start);
543  scheduled_point_t *o_pt =
544  o.m_sched_point_tree.get_state (o.m_plan_start);
545  while (this_pt) {
546  if (*this_pt != *o_pt)
547  return false;
548  this_pt = m_sched_point_tree.next (this_pt);
549  o_pt = o.m_sched_point_tree.next (o_pt);
550  }
551  }
552  return true;
553 }
