140 void test_match ()
141 {
142  resource resource;
143  resource.add_property ("xx");
144  resource.add_property ("yy");
145  resource.type = "node";
146  resource.name = "foo0";
147  resource.basename = "foo";
148  resource.rank = 0;
149 
150  struct match_test *t = match_tests;
151  while (t->desc) {
152  auto c = constraint_parser (YAML::Load (t->json));
153  errno = 0;
154  ok (c->match (resource) == t->result, "%s", t->desc);
155  ok (errno == 0, "errno is preserved");
156  t++;
157  }
158 }

262 void test_validate ()
263 {
264  struct validate_test *t = validate_tests;
265  while (t->desc) {
266  bool exception = false;
267  std::string errmsg;
268 
269  try {
270  constraint_parser (YAML::Load (t->json));
271  }
272  catch (Flux::Jobspec::parse_error& e) {
273  exception = true;
274  errmsg = e.what ();
275  }
276  if (t->rc != 0) {
277  ok (exception, "%s", t->desc);
278  is (errmsg.c_str(), t->err, "%s: got expected error", t->desc);
279  }
280  else {
281  ok (!exception, "%s", t->desc);
282  }
283  t++;
284  }
285 }
