40 extern "C" planner_multi_t *planner_multi_new (
41  int64_t base_time, uint64_t duration,
42  const uint64_t *resource_totals,
43  const char **resource_types, size_t len)
44 {
45  size_t i = 0;
46  planner_multi_t *ctx = nullptr;
47 
48  if (duration < 1 || !resource_totals || !resource_types) {
49  errno = EINVAL;
50  goto error;
51  } else {
52  for (i = 0; i < len; ++i) {
53  if (resource_totals[i] >
54  static_cast<uint64_t> (std::numeric_limits<int64_t>::max ())) {
55  errno = ERANGE;
56  goto error;
57  }
58  }
59  }
60 
61  try {
62  ctx = new planner_multi_t (base_time, duration, resource_totals,
63  resource_types, len);
64  } catch (std::bad_alloc &e) {
65  goto nomem_error;
66  }
67  return ctx;
68 
69 nomem_error:
70  errno = ENOMEM;
71  planner_multi_destroy (&ctx);
72 error:
73  return ctx;

223 extern "C" int64_t planner_multi_avail_time_first (
224  planner_multi_t *ctx, int64_t on_or_after,
225  uint64_t duration,
226  const uint64_t *resource_requests, size_t len)
227 {
228  size_t i = 0;
229  int unmet = 0;
230  int64_t t = -1;
231 
232  if (!ctx || !resource_requests || ctx->plan_multi->get_planners_size () < 1
233  || ctx->plan_multi->get_planners_size () != len) {
234  errno = EINVAL;
235  goto done;
236  }
237 
238  fill_iter_request (ctx, &(ctx->plan_multi->get_iter ()),
239  on_or_after, duration, resource_requests, len);
240 
241  if ((t = planner_avail_time_first (ctx->plan_multi->get_planner_at (
242  static_cast<size_t> (0)), on_or_after,
243  duration, resource_requests[0])) == -1)
244  goto done;
245 
246  do {
247  unmet = 0;
248  for (i = 1; i < ctx->plan_multi->get_planners_size (); ++i) {
249  if ((unmet = planner_avail_during (ctx->plan_multi->get_planner_at (i),
250  t, duration,
251  resource_requests[i])) == -1)
252  break;
253  }
254  } while (unmet && (t = planner_avail_time_next (
255  ctx->plan_multi->get_planner_at (static_cast<size_t> (0)))) != -1);
256 
257 done:
258  return t;
259 }

262 {

359 extern "C" int64_t planner_multi_add_span (
360  planner_multi_t *ctx, int64_t start_time,
361  uint64_t duration,
362  const uint64_t *resource_requests, size_t len)
363 {
364  size_t i = 0;
365  int64_t span = -1;
366  int64_t mspan = -1;
367 
368  if (!ctx || !resource_requests || len != ctx->plan_multi->get_planners_size ())
369  return -1;
370 
371  mspan = ctx->plan_multi->get_span_counter ();
372  auto res = ctx->plan_multi->get_span_lookup ().insert (
373  std::pair<int64_t, std::vector<int64_t>> (
374  mspan, std::vector<int64_t> ()));
375  if (!res.second) {
376  errno = EEXIST;
377  return -1;
378  }
379 
380  ctx->plan_multi->incr_span_counter ();
381 
382  for (i = 0; i < len; ++i) {
383  if ( (span = planner_add_span (ctx->plan_multi->get_planner_at (i),
384  start_time, duration,
385  resource_requests[i])) == -1) {
386  ctx->plan_multi->get_span_lookup ().erase (mspan);
387  return -1;
388  }
389  ctx->plan_multi->get_span_lookup ()[mspan].push_back (span);
390  }
391  return mspan;
392 }

470 extern "C" int planner_multi_update (planner_multi_t *ctx,
471  const uint64_t *resource_totals,
472  const char **resource_types,
473  size_t len)
474 {
475  int rc = -1;
476  size_t i = 0;
477  // Assuming small number of resource types,
478  // could try set, too
479  std::unordered_set<std::string> rtypes;
480  int64_t base_time = 0;
481  int64_t duration = 0;
482 
483  if (!ctx || !resource_totals || !resource_types) {
484  errno = EINVAL;
485  goto done;
486  }
487  base_time = planner_base_time (
488  ctx->plan_multi->get_planner_at (static_cast<size_t> (0)));
489  duration = planner_duration (
490  ctx->plan_multi->get_planner_at (static_cast<size_t> (0)));
491  if (duration < 0) {
492  errno = EINVAL;
493  goto done;
494  }
495 
496  for (i = 0; i < len; ++i) {
497  if (resource_totals[i] >
498  static_cast<uint64_t> (std::numeric_limits<int64_t>::max ())) {
499  errno = ERANGE;
500  goto done;
501  }
502  rtypes.insert (resource_types[i]);
503  if (!ctx->plan_multi->planner_at (resource_types[i])) {
504  // Assume base_time same as parent
505  ctx->plan_multi->add_planner (base_time, static_cast<uint64_t> (duration),
506  resource_totals[i], resource_types[i], i);
507  } else {
508  // Index could have changed
509  ctx->plan_multi->update_planner_index (resource_types[i], i);
510  if ( (rc = ctx->plan_multi->update_planner_total (resource_totals[i],
511  i)) != 0) {
512  errno = EINVAL;
513  goto done;
514  }
515  }
516  }
517  // remove values not in new types
518  if (rtypes.size () > 0)
519  ctx->plan_multi->delete_planners (rtypes);
520 
521  rc = 0;
522 
523 done:
524  return rc;
525 }
