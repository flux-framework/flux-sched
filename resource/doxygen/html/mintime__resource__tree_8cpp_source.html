29 int64_t mintime_resource_tree_t::right_branch_mintime (mt_resource_rb_node_t *n) const
30 {
31  int64_t min_time = std::numeric_limits<int64_t>::max ();
32  mt_resource_rb_node_t *right = n->get_right ();
33  if (right)
34  min_time = right->subtree_min;
35  return (n->at < min_time)? n->at : min_time;
36 }

38 scheduled_point_t *mintime_resource_tree_t::find_mintime_point (
39  mt_resource_rb_node_t *anchor, int64_t min_time) const
40 {
41  if (!anchor)
42  return nullptr;
43 
44  if (anchor->at == min_time)
45  return anchor->get_point ();
46 
47  mt_resource_rb_node_t *node = anchor->get_right ();
48  while (node) {
49  if (node->at == min_time)
50  return node->get_point ();
51  if (node->get_left ()
52  && (node->get_left ()->subtree_min == min_time))
53  node = node->get_left ();
54  else
55  node = node->get_right ();
56  }
57 
58  // Error condition: when an anchor was found, there must be
59  // a point that meets the requirements.
60  errno = ENOTSUP;
61  return nullptr;
62 }

64 int64_t mintime_resource_tree_t::find_mintime_anchor (
65  int64_t request, mt_resource_rb_node_t **anchor_p) const
66 {
67  mt_resource_rb_node_t *node = m_tree.get_root ();
68  int64_t min_time = std::numeric_limits<int64_t>::max ();
69  int64_t right_min_time = std::numeric_limits<int64_t>::max ();
70  while (node) {
71  if (request <= node->remaining) {
72  // visiting node satisfies the resource requirements. This means all
73  // nodes at its subtree also satisfy the requirements. Thus,
74  // right_min_time is the best min time.
75  right_min_time = right_branch_mintime (node);
76  if (right_min_time < min_time) {
77  min_time = right_min_time;
78  *anchor_p = node;
79  }
80  // next, we should search the left subtree for potentially better
81  // then current min_time;
82  node = node->get_left ();
83  } else {
84  // visiting node does not satisfy the resource requirements. This
85  // means that nothing in its left branch will meet these requirements:
86  // time to search the right subtree.
87  node = node->get_right ();
88  }
89  }
90  return min_time;
91 }

100 template <class mt_resource_rb_node_t, class NodeTraits>
101 void mt_resource_node_traits<mt_resource_rb_node_t, NodeTraits>::fix (
102  mt_resource_rb_node_t *node)
103 {
104  mt_resource_rb_node_t *trav = node;
105  while (trav) {
106  int64_t old = trav->subtree_min;
107  trav->subtree_min = trav->at;
108  mt_resource_rb_node_t *left = trav->get_left ();
109  if (left) {
110  if (trav->subtree_min > left->subtree_min)
111  trav->subtree_min = left->subtree_min;
112  }
113  mt_resource_rb_node_t *right = trav->get_right ();
114  if (right) {
115  if (trav->subtree_min > right->subtree_min)
116  trav->subtree_min = right->subtree_min;
117  }
118  // no parent propagation is needed if new subtree_min == old,
119  if (trav->subtree_min == old)
120  break;
121 
122  mt_resource_rb_node_t *parent = trav->get_parent ();
123  // if parent's subtree_min is already smaller than
124  // new subtree_min, no parent propagation is necessary.
125  // However, if parent's subtree_min == old, propagation is
126  // needed, as it could (or in fact is likely to)
127  // come from the traversing child node and thus needs an update.
128  if (parent && parent->subtree_min != old
129  && parent->subtree_min < trav->subtree_min)
130  break;
131  trav = parent;
132  }
133 }
