84 void match_allocate_multi_cont (flux_future_t *f, void *arg)
85 {
86  int64_t rj = -1;
87  int64_t at;
88  double ov;
89  const char *rset = nullptr;
90  const char *status = nullptr;
91  queue_adapter_base_t *adapter = static_cast<queue_adapter_base_t *> (arg);
92 
93  if (flux_rpc_get_unpack (f, "{s:I s:s s:f s:s s:I}",
94  "jobid", &rj,
95  "status", &status,
96  "overhead", &ov,
97  "R", &rset,
98  "at", &at) == 0) {
99  if (adapter->handle_match_success (rj, status, rset, at, ov) < 0) {
100  adapter->set_sched_loop_active (false);
101  flux_future_destroy (f);
102  return;
103  }
104  } else {
105  adapter->handle_match_failure (errno);
106  adapter->set_sched_loop_active (false);
107  flux_future_destroy (f);
108  return;
109  }
110  flux_future_reset (f);
111  return;
112 }

114 int reapi_module_t::match_allocate_multi (void *h,
115  match_op_t match_op,
116  const char *jobs,
117  queue_adapter_base_t *adapter)
118 {
119  int rc = -1;
120  flux_t *fh = static_cast<flux_t *> (h);
121  flux_future_t *f = nullptr;
122  const char *cmd = match_op_to_string (match_op);
123 
124  if (!fh) {
125  errno = EINVAL;
126  goto error;
127  }
128  if ( !(f = flux_rpc_pack (fh,
129  "sched-fluxion-resource.match_multi",
130  FLUX_NODEID_ANY, FLUX_RPC_STREAMING,
131  "{s:s s:s}",
132  "cmd", cmd,
133  "jobs", jobs)))
134  goto error;
135  if (flux_future_then (f,
136  -1.0f,
137  match_allocate_multi_cont,
138  static_cast<void *> (adapter)) < 0)
139  goto error;
140  return 0;
141 
142 error:
143  flux_future_destroy (f);
144  return rc;

158 int reapi_module_t::update_allocate (void *h, const uint64_t jobid,
159  const std::string &R, int64_t &at,
160  double &ov, std::string &R_out)
161 {
162  int rc = -1;
163  int64_t rj = -1;
164  flux_t *fh = (flux_t *)h;
165  flux_future_t *f = NULL;
166  int64_t scheduled_at = -1;
167  double overhead = 0.0f;
168  const char *rset = NULL;
169  const char *status = NULL;
170 
171  if (!fh || R == "" || jobid > INT64_MAX) {
172  errno = EINVAL;
173  goto out;
174  }
175  if ( !(f = flux_rpc_pack (fh, "sched-fluxion-resource.update",
176  FLUX_NODEID_ANY, 0,
177  "{s:I s:s}",
178  "jobid", jobid,
179  "R", R.c_str ())))
180  goto out;
181  if ( (rc = flux_rpc_get_unpack (f, "{s:I s:s s:f s:s s:I}",
182  "jobid", &rj,
183  "status", &status,
184  "overhead", &overhead,
185  "R", &rset,
186  "at", &scheduled_at)) < 0)
187  goto out;
188  if (rj != static_cast<int64_t> (jobid)
189  || rset == NULL
190  || status == NULL
191  || std::string ("ALLOCATED") != status) {
192  rc = -1;
193  errno = EPROTO;
194  goto out;
195  }
196  R_out = rset;
197  ov = overhead;
198  at = scheduled_at;
199 
200 out:
201  flux_future_destroy (f);
202  return rc;
203 }

205 int reapi_module_t::cancel (void *h, const uint64_t jobid, bool noent_ok)
206 {
207  int rc = -1;
208  flux_t *fh = (flux_t *)h;
209  flux_future_t *f = NULL;
210  int saved_errno;
211 
212  if (!fh || jobid > INT64_MAX) {
213  errno = EINVAL;
214  goto out;
215  }
216  if (!(f = flux_rpc_pack (fh, "sched-fluxion-resource.cancel",
217  FLUX_NODEID_ANY, 0,
218  "{s:I}", "jobid", (const int64_t)jobid))) {
219  goto out;
220  }
221  saved_errno = errno;
222  if ((rc = flux_rpc_get (f, NULL)) < 0) {
223  if (noent_ok && errno == ENOENT) {
224  errno = saved_errno;
225  rc = 0;
226  }
227  goto out;
228  }
229  rc = 0;
230 
231 out:
232  flux_future_destroy (f);
233  return rc;
