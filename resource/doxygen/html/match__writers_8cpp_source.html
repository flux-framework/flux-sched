37 int match_writers_t::compress_ids (std::stringstream &o,
38  const std::vector<int64_t> &ids)
39 {
40  int rc = 0;
41  try {
42  int64_t base = INT64_MIN;
43  int64_t runlen = 0;
44 
45  for (auto &id : ids) {
46  if (id == (base + runlen + 1)) {
47  runlen++;
48  } else {
49  if (runlen != 0)
50  o << "-" + std::to_string (base + runlen) + ",";
51  else if (base != INT64_MIN)
52  o << ",";
53  o << id;
54  base = id;
55  runlen = 0;
56  }
57  }
58  if (runlen)
59  o << "-" << (base + runlen);
60  } catch (std::bad_alloc &) {
61  rc = -1;
62  errno = ENOMEM;
63  }
64  return rc;
65 }

84  } else {

86  rc = -1;

93  }
94 
95  if ( !(*hostlist_out = hostlist_encode (hl))) {
96  rc = -1;
97  goto ret;
98  }
99

403 json_t *jgf_match_writers_t::emit_vtx_base (const f_resource_graph_t &g,
404  const vtx_t &u,
405  unsigned int needs, bool exclusive)
406 {
407  json_t *o = NULL;
408  json_t *prop = nullptr;
409 
410  if (!g[u].properties.empty ()) {
411  if ( !(prop = json_object ())) {
412  errno = ENOMEM;
413  return nullptr;
414  }
415  for (auto &kv : g[u].properties) {
416  json_t *value = nullptr;
417  if ( !(value = json_string (kv.second.c_str ()))) {
418  json_decref (prop);
419  errno = EINVAL;
420  return nullptr;
421  }
422  if (json_object_set_new (prop, kv.first.c_str (), value) < 0) {
423  json_decref (prop);
424  errno = EINVAL;
425  return nullptr;
426  }
427  }
428  }
429 
430  if (prop) {
431  if (!(o = json_pack ("{s:s s:s s:s s:I s:I s:i s:o s:b s:s s:I}",
432  "type", g[u].type.c_str (),
433  "basename", g[u].basename.c_str (),
434  "name", g[u].name.c_str (),
435  "id", g[u].id,
436  "uniq_id", g[u].uniq_id,
437  "rank", g[u].rank,
438  "properties", prop,
439  "exclusive", (exclusive)? 1 : 0,
440  "unit", g[u].unit.c_str (),
441  "size", static_cast<int64_t> (needs)))) {
442  errno = EINVAL;
443  }
444  } else {
445  if (!(o = json_pack ("{s:s s:s s:s s:I s:I s:i s:b s:s s:I}",
446  "type", g[u].type.c_str (),
447  "basename", g[u].basename.c_str (),
448  "name", g[u].name.c_str (),
449  "id", g[u].id,
