183 int resource_gen_spec_t::write_graphviz (const std::string &ofn, bool simple)
184 {
185  int rc = 0;
186  std::fstream out_file (ofn, std::fstream::out);
187  try {
188  vtx_basename_map_t v_bn_map = get (&resource_pool_gen_t::basename, g);
189  vtx_size_map_t v_sz_map = get (&resource_pool_gen_t::size, g);
190  //vtx_unit_map_t v_ut_map = get (&resource_pool_gen_t::unit, g);
191  vtx_subsystem_map_t v_ss_map = get (&resource_pool_gen_t::subsystem, g);
192  edg_relation_map_t e_rel_map = get (&relation_gen_t::relation, g);
193  edg_gen_method_map_t e_gen_map = get (&relation_gen_t::gen_method, g);
194  edg_multi_scale_map_t e_ms_map = get (&relation_gen_t::multi_scale, g);
195  if (!simple) {
196  gg_label_writer_t<vtx_subsystem_map_t,
197  vtx_basename_map_t, vtx_size_map_t> vwr (
198  v_ss_map, v_bn_map, v_sz_map);
199  gg_label_writer_t<edg_relation_map_t,
200  edg_gen_method_map_t, edg_multi_scale_map_t> ewr (
201  e_rel_map, e_gen_map, e_ms_map);
202  boost::write_graphviz (out_file, g, vwr, ewr);
203  } else {
204  gg_label_writer_sim_t<vtx_basename_map_t> vwr (v_bn_map);
205  gg_label_writer_sim_t<edg_relation_map_t> ewr (e_rel_map);
206  boost::write_graphviz (out_file, g, vwr, ewr);
207  }
208  } catch (boost::graph_exception &e) {
209  rc = -1;
210  }
211  out_file.close ();
212  return rc;
213 }
