52 int var_aware_t::dom_finish_graph (
53  const subsystem_t &subsystem,
54  const std::vector<Flux::Jobspec::Resource> &resources,
55  const f_resource_graph_t &g, scoring_api_t &dfu)
56 {
57  int score = MATCH_MET;
58  fold::less comp;
59 
60  for (auto &resource : resources) {
61  const std::string &type = resource.type;
62  unsigned int qc = dfu.qualified_count (subsystem, type);
63  unsigned int count = calc_count (resource, qc);
64  if (count == 0) {
65  score = MATCH_UNMET;
66  break;
67  }
68  dfu.choose_accum_best_k (subsystem, type, count, comp);
69  }
70  dfu.set_overall_score (score);
71  return (score == MATCH_MET)? 0 : -1;
72 }

85 int var_aware_t::dom_finish_vtx (
86  vtx_t u,
87  const subsystem_t &subsystem,
88  const std::vector<Flux::Jobspec::Resource> &resources,
89  const f_resource_graph_t &g, scoring_api_t &dfu)
90 {
91  int64_t score = MATCH_MET;
92  int64_t overall;
93  fold::less comp;
94 
95  /* Default value for worst-performing-class assumed as 9999. */
96  int64_t perf_class = 9999;
97 
98  for (auto &resource : resources) {
99  if (resource.type != g[u].type)
100  continue;
101 
102  // jobspec resource type matches with the visiting vertex
103  for (auto &c_resource : resource.with) {
104  // test children resource count requirements
105  const std::string &c_type = c_resource.type;
106  unsigned int qc = dfu.qualified_count (subsystem, c_type);
107  unsigned int count = calc_count (c_resource, qc);
108  if (count == 0) {
109  score = MATCH_UNMET;
110  break;
111  }
112  dfu.choose_accum_best_k (subsystem, c_resource.type, count, comp);
113  }
114  }
115 
116  if (score == MATCH_MET) {
117  try {
118  perf_class = std::stoi (g[u].properties.at ("perf_class"));
119  }
120  catch (const std::out_of_range& oor) {
121  perf_class = 9999;
122  }
123  catch (const std::invalid_argument& ia) {
124  perf_class = 9999;
125  }
126  /* Ensure we don't have negative or more than
127  * 9999 performance classes so we have clear boundaries. */
128  if (perf_class < 0 || perf_class > 9999) {
129  perf_class = 9999;
130  }
131  }
132 
133  overall = (score == MATCH_MET)? (score + perf_class) : score;
134  dfu.set_overall_score (overall);
135  decr ();
136  return (score == MATCH_MET)? 0 : -1;
137 }
